47 RefreshIconEngine::RefreshIconEngine()
48 {
49  // We test if a QApplication object exists. While the constructor
50  // would work without problems also when there is no QApplication
51  // object, this is not true for the member functions: Each instanciation
52  // of QPixmap and each call to the (static!) QGuiApplication::palette()
53  // would lead to a crash. And the paint function also requires a
54  // QApplication object. This is confusing, so it seems better to
55  // crash directly here, with a useful error message. (Note that also
56  // QWidget crashes in the constructor when no QApplication object is
57  // available.)
58  if (qobject_cast<QApplication *>(QCoreApplication::instance()) == nullptr) {
59  // From Qt documentation about qFatal():
60  //
61  // “If you are using the default message handler
62  // this function will abort to create a core dump.”
63  //
64  // So, it should not be necessary to throw an exception after
65  // calling qFatal, because qFatal itself aborts the program.
66  qFatal(
67  "RefreshIconEngine: Must construct a QApplication "
68  "before a RefreshIconEngine");
69  }
70 }

115 QPixmap RefreshIconEngine::pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state)
116 {
117  QImage iconImage {size, QImage::Format_ARGB32_Premultiplied};
118  iconImage.fill(Qt::transparent);
119  {
120  // Putting this in an own block, as the QPainter object
121  // might be in an undefined state after paint() has been
122  // called. So the QPainter object should not be used
123  // anymore after the paint() call.
124  QPainter painter(&iconImage);
125  paint(&painter, QRect(QPoint(0, 0), size), mode, state);
126  }
127  return QPixmap::fromImage(iconImage);
128 }

140 void RefreshIconEngine::paintFallbackIcon(QPainter *painter, const QRect rect, QIcon::Mode mode)
141 {
142  if (rect.isEmpty()) {
143  // Return on empty rectangles. This avoids bad calculations with
144  // formulas that require a non-empty, non-negative (!) size.
145  return;
146  }
147 
148  // It might be possible that rect.width() and rect.height are not
149  // identical. Propably the best thing to do would be to center
150  // the (square) icon within this rectangle. On the other side,
151  // Qt’s own implementation does’t to that either, but simply
152  // aligns to the center of the coordinate system (top left corner),
153  // also on right-to-left locales. So, for simplicity, we do the same
154  // thing here.
155  constexpr int unscaledPenWidth = 1;
156  constexpr int designSpace = 16;
157  const int destinationSpace = qMin(rect.width(), rect.height());
158  const qreal halfDestinationSpace = destinationSpace / 2.0;
159  const qreal scaleFactor = static_cast<qreal>(destinationSpace) / designSpace;
160  const qreal radius = halfDestinationSpace - 0.5 * unscaledPenWidth * scaleFactor;
161  painter->setRenderHint(QPainter::RenderHint::Antialiasing);
162  painter->setCompositionMode(
163  // While CompositionMode_SourceOver is the default value
164  // anyway, it’s important to set it explicitly, as the
165  // painter that we received might currently be set to
166  // another CopoisitionMode.
167  QPainter::CompositionMode::CompositionMode_SourceOver);
168  QPen pen;
169  pen.setWidthF(unscaledPenWidth * scaleFactor);
170  QPalette::ColorGroup paletteColorGroup;
171  switch (mode) {
172  case QIcon::Mode::Normal:
173  paletteColorGroup = QPalette::ColorGroup::Normal;
174  break;
175  case QIcon::Mode::Disabled:
176  paletteColorGroup = QPalette::ColorGroup::Disabled;
177  break;
178  case QIcon::Mode::Active:
179  paletteColorGroup = QPalette::ColorGroup::Active;
180  break;
181  case QIcon::Mode::Selected:
182  paletteColorGroup = QPalette::ColorGroup::Normal;
183  break;
184  }
185  QPalette referencePalette;
186  if (m_referenceWidget.isNull()) {
187  referencePalette = QGuiApplication::palette();
188  } else {
189  referencePalette = m_referenceWidget->palette();
190  }
191  pen.setColor(referencePalette.color(paletteColorGroup, QPalette::ColorRole::Text));
192  painter->setPen(pen);
193  painter->drawEllipse(QPointF(halfDestinationSpace + rect.left(), halfDestinationSpace + rect.top()), radius, radius);
194 }

204 void RefreshIconEngine::paint(QPainter *painter, const QRect &rect, QIcon::Mode mode, QIcon::State state)
205 {
206  // Initialize
207  QIcon myIcon;
208 
209  // First, try to load an icon from the current icon theme.
210  myIcon = QIcon::fromTheme(QStringLiteral("dialog-apply"));
211  if (!myIcon.isNull()) {
212  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
213  return;
214  }
215  myIcon = QIcon::fromTheme(QStringLiteral("gtk-apply"));
216  if (!myIcon.isNull()) {
217  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
218  return;
219  }
220  myIcon = QIcon::fromTheme(QStringLiteral("dialog-ok-apply"));
221  if (!myIcon.isNull()) {
222  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
223  return;
224  }
225  myIcon = QIcon::fromTheme(QStringLiteral("gtk-ok-apply"));
226  if (!myIcon.isNull()) {
227  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
228  return;
229  }
230  myIcon = QIcon::fromTheme(QStringLiteral("dialog-ok"));
231  if (!myIcon.isNull()) {
232  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
233  return;
234  }
235  myIcon = QIcon::fromTheme(QStringLiteral("gtk-ok"));
236  if (!myIcon.isNull()) {
237  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
238  return;
239  }
240 
241  // Sedond, if the current icon theme has no icon (not all platforms
242  // provide icon themes, and even those who provide it might be incomplete)
243  // then try to use a build-in icon from the current widget style.
244  // The q_pointer’s object is still not fully initialized at
245  // this point, but it’s base class constructor has fully run;
246  // this should be enough to use functionality based on QWidget.
247  QStyle *referenceStyle;
248  if (m_referenceWidget.isNull()) {
249  // We can use QApplication::style(), which requires (in spite of
250  // being static), that a QApplication object existes). For an existing
251  // QApplication object is checked in the constructor of this class.
252  referenceStyle = QApplication::style();
253  } else {
254  referenceStyle = m_referenceWidget->style();
255  }
256  myIcon = referenceStyle->standardIcon(QStyle::StandardPixmap::SP_DialogApplyButton);
257  if (!myIcon.isNull()) {
258  myIcon.paint(painter, rect, Qt::AlignCenter, mode, state);
259  return;
260  }
261 
262  // Third, if no other option has returned a valid icon, we use
263  // our build-in icon.
264  paintFallbackIcon(painter, rect, mode);
265 }
