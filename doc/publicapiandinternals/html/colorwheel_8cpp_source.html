55 ColorWheel::ColorWheel(const QSharedPointer<PerceptualColor::RgbColorSpace> &colorSpace, QWidget *parent)
56  : AbstractDiagram(parent)
57  , d_pointer(new ColorWheelPrivate(this, colorSpace))
58 {
59  // Setup the color space must be the first thing to do because
60  // other operations rely on a working color space.
61  d_pointer->m_rgbColorSpace = colorSpace;
62 
63  // Initialization
64  d_pointer->m_hue = LchValues::neutralHue;
65 
66  // Set focus policy
67  // In Qt, usually focus (QWidget::hasFocus()) by mouse click is
68  // either not accepted at all or accepted always for the hole rectangular
69  // widget, depending on QWidget::focusPolicy(). This is not
70  // convenient and intuitive for big, circular-shaped widgets like this one.
71  // It would be nicer if the focus would only be accepted by mouse clicks
72  // <em>within the circle itself</em>. Qt does not provide a build-in way to
73  // do this. But a workaround to implement this behavior is possible: Set
74  // QWidget::focusPolicy() to <em>not</em> accept focus by mouse
75  // click. Then, reimplement mousePressEvent() and call
76  // setFocus(Qt::MouseFocusReason) if the mouse click is within the
77  // circle. Therefore, this class simply defaults to
78  // Qt::FocusPolicy::TabFocus for QWidget::focusPolicy().
79  setFocusPolicy(Qt::FocusPolicy::TabFocus);
80 }

127 {

149 void ColorWheel::mousePressEvent(QMouseEvent *event)
150 {
151  const qreal radius = maximumWidgetSquareSize() / 2.0 - spaceForFocusIndicator();
152  PolarPointF myPolarPoint = d_pointer->fromWidgetToWheelCoordinates(event->pos());
153 
154  // Ignare clicks outside the wheel
155  if (myPolarPoint.radial() > radius) {
156  // Make sure default coordinates like drag-window
157  // in KDE’s Breeze widget style works:
158  event->ignore();
159  return;
160  }
161 
162  // If inside the wheel (either in the wheel ribbon itself or in the hole
163  // in the middle), take focus:
164  setFocus(Qt::MouseFocusReason);
165 
166  if (myPolarPoint.radial() > radius - gradientThickness()) {
167  d_pointer->m_isMouseEventActive = true;
168  setHue(myPolarPoint.angleDegree());
169  } else {
170  // Make sure default coordinates like drag-window
171  // in KDE’s Breeze widget style works:
172  event->ignore();
173  }
174 
175  return;
176 }

234 void ColorWheel::wheelEvent(QWheelEvent *event)
235 {
236  const qreal radius = maximumWidgetSquareSize() / 2.0 - spaceForFocusIndicator();
237  PolarPointF myPolarPoint = d_pointer->fromWidgetToWheelCoordinates(event->pos());
238  if (
239  // Do nothing while mouse movement is tracked anyway. This would
240  // be confusing:
241  (!d_pointer->m_isMouseEventActive)
242  // Only react on wheel events when its in the wheel ribbon or in
243  // the inner hole:
244  && (myPolarPoint.radial() <= radius)
245  // Only react on good old vertical wheels, and not on horizontal wheels:
246  && (event->angleDelta().y() != 0)
247  // then:
248  ) {
249  d_pointer->setHueNormalized(d_pointer->m_hue + standardWheelStepCount(event) * singleStepHue);
250  } else {
251  event->ignore();
252  }
253 }

269 void ColorWheel::keyPressEvent(QKeyEvent *event)
270 {
271  switch (event->key()) {
272  case Qt::Key_Plus:
273  d_pointer->setHueNormalized(d_pointer->m_hue + singleStepHue);
274  break;
275  case Qt::Key_Minus:
276  d_pointer->setHueNormalized(d_pointer->m_hue - singleStepHue);
277  break;
278  case Qt::Key_Insert:
279  d_pointer->setHueNormalized(d_pointer->m_hue + pageStepHue);
280  break;
281  case Qt::Key_Delete:
282  d_pointer->setHueNormalized(d_pointer->m_hue - pageStepHue);
283  break;
284  default:
285  /* Quote from Qt documentation:
286  *
287  * If you reimplement this handler, it is very important
288  * that you call the base class implementation if you do not
289  * act upon the key.
290  *
291  * The default implementation closes popup widgets if the user
292  * presses the key sequence for QKeySequence::Cancel (typically
293  * the Escape key). Otherwise the event is ignored, so that the
294  * widget’s parent can interpret it. */
295  QWidget::keyPressEvent(event);
296  break;
297  }
298 }

318 void ColorWheel::paintEvent(QPaintEvent *event)
319 {
320  Q_UNUSED(event);
321  // We do not paint directly on the widget, but on a QImage buffer first:
322  // Render anti-aliased looks better. But as Qt documentation says:
323  //
324  // “Renderhints are used to specify flags to QPainter that may or
325  // may not be respected by any given engine.”
326  //
327  // Painting here directly on the widget might lead to different
328  // anti-aliasing results depending on the underlying window system. This
329  // is especially problematic as anti-aliasing might shift or not a pixel
330  // to the left or to the right. So we paint on a QImage first. As QImage
331  // (at difference to QPixmap and a QWidget) is independent of native
332  // platform rendering, it guarantees identical anti-aliasing results on
333  // all platforms. Here the quote from QPainter class documentation:
334  //
335  // “To get the optimal rendering result using QPainter, you should
336  // use the platform independent QImage as paint device; i.e. using
337  // QImage will ensure that the result has an identical pixel
338  // representation on any platform.”
339  QImage paintBuffer(maximumPhysicalSquareSize(), // width
340  maximumPhysicalSquareSize(), // height
341  QImage::Format_ARGB32_Premultiplied // format
342  );
343  paintBuffer.fill(Qt::transparent);
344  paintBuffer.setDevicePixelRatio(devicePixelRatioF());
345  QPainter bufferPainter(&paintBuffer);
346 
347  // Paint the color wheel
348  bufferPainter.setRenderHint(QPainter::Antialiasing, false);
349  // As devicePixelRatioF() might have changed, we make sure everything
350  // that might depend on devicePixelRatioF() is updated before painting.
351  d_pointer->m_wheelImage.setBorder(spaceForFocusIndicator() * devicePixelRatioF());
352  d_pointer->m_wheelImage.setDevicePixelRatioF(devicePixelRatioF());
353  d_pointer->m_wheelImage.setImageSize(maximumPhysicalSquareSize());
354  d_pointer->m_wheelImage.setWheelThickness(gradientThickness() * devicePixelRatioF());
355  bufferPainter.drawImage(QPoint(0, 0), // image position (top-left)
356  d_pointer->m_wheelImage.getImage() // the image itself
357  );
358 
359  // Paint the handle
360  const qreal wheelOuterRadius = maximumWidgetSquareSize() / 2.0 - spaceForFocusIndicator();
361  // Get widget coordinates for the handle
362  QPointF myHandleInner = d_pointer->fromWheelToWidgetCoordinates(
363  // Inner point at the wheel:
364  PolarPointF(wheelOuterRadius - gradientThickness(), // x
365  d_pointer->m_hue // y
366  ));
367  QPointF myHandleOuter = d_pointer->fromWheelToWidgetCoordinates(
368  // Outer point at the wheel:
369  PolarPointF(wheelOuterRadius, d_pointer->m_hue));
370  // Draw the line
371  QPen pen;
372  pen.setWidth(handleOutlineThickness());
373  pen.setCapStyle(Qt::FlatCap);
374  pen.setColor(Qt::black);
375  bufferPainter.setPen(pen);
376  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
377  bufferPainter.drawLine(myHandleInner, myHandleOuter);
378 
379  // Paint a focus indicator if the widget has the focus
380  if (hasFocus()) {
381  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
382  pen = QPen();
383  pen.setWidth(handleOutlineThickness());
384  pen.setColor(focusIndicatorColor());
385  bufferPainter.setPen(pen);
386  const qreal center = maximumWidgetSquareSize() / 2.0;
387  bufferPainter.drawEllipse(
388  // center:
389  QPointF(center, center),
390  // x radius:
391  center - handleOutlineThickness() / 2.0,
392  // y radius:
393  center - handleOutlineThickness() / 2.0);
394  }
395 
396  // Paint the buffer to the actual widget
397  QPainter widgetPainter(this);
398  widgetPainter.setRenderHint(QPainter::Antialiasing, false);
399  widgetPainter.drawImage(QPoint(0, 0), paintBuffer);
400 }

466 QSize ColorWheel::minimumSizeHint() const
467 {
468  // We interpretate the gradientMinimumLength() as the length between two
469  // poles of human perception. Around the wheel, there are four of them
470  // (0° red, 90° yellow, 180° green, 270° blue). So the circumference of
471  // the inner circle of the wheel is 4 × gradientMinimumLength(). By
472  // dividing it by π, we get the requiered inner diameter:
473  const qreal innerDiameter = 4 * gradientMinimumLength() / M_PI;
474  const int size = qRound(innerDiameter + 2 * gradientThickness() + 2 * spaceForFocusIndicator());
475  // Expand to the global minimum size for GUI elements
476  return QSize(size, size).expandedTo(QApplication::globalStrut());
477 }
