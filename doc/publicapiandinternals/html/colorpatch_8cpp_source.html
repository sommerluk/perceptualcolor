118 void ColorPatch::paintEvent(QPaintEvent *event)
119 {
120  // First of all, draw the frame
121  QFrame::paintEvent(event);
122 
123  // Initialization
124  QPainter widgetPainter {this};
125  widgetPainter.setRenderHint(QPainter::Antialiasing, true);
126  QStyleOptionFrame opt;
127  opt.initFrom(this); // Sets also QStyle::State_MouseOver if appropriate
128  const int defaultFrameWidth = qMax(style()->pixelMetric(QStyle::PM_DefaultFrameWidth, &opt), 1);
129 
130  // Draw content of an invalid color (and return)
131  if (!d_pointer->m_color.isValid()) {
132  widgetPainter.setRenderHint(QPainter::Antialiasing, true);
133  QPalette::ColorGroup paletteColorGroup;
134  if (isEnabled()) {
135  paletteColorGroup = QPalette::ColorGroup::Normal;
136  } else {
137  paletteColorGroup = QPalette::ColorGroup::Disabled;
138  }
139  QPen pen {palette().color(paletteColorGroup, QPalette::WindowText)};
140  pen.setWidth(defaultFrameWidth);
141  pen.setCapStyle(Qt::PenCapStyle::RoundCap);
142  widgetPainter.setPen(pen);
143  // Calculate an adjustmend to allow for rounded line caps without
144  // touching the frame.
145  const qreal adjustmentForRoundedLineCaps = pen.widthF() / 2.0;
146  const QRectF noColorMarkerRectangle = QRectF(contentsRect()).adjusted(adjustmentForRoundedLineCaps, adjustmentForRoundedLineCaps, -adjustmentForRoundedLineCaps, -adjustmentForRoundedLineCaps);
147  widgetPainter.drawLine(noColorMarkerRectangle.topLeft(), noColorMarkerRectangle.bottomRight());
148  widgetPainter.drawLine(noColorMarkerRectangle.topRight(), noColorMarkerRectangle.bottomLeft());
149  return;
150  }
151 
152  // Draw content of a valid color
153  widgetPainter.setRenderHint(QPainter::Antialiasing, false);
154  // Create an image to paint on
155  QImage tempImage = QImage(
156  // deliberately rounding down with static_cast<int>
157  static_cast<int>(contentsRect().width() * devicePixelRatioF()),
158  static_cast<int>(contentsRect().height() * devicePixelRatioF()),
159  QImage::Format_RGB32);
160  if (d_pointer->m_color.alphaF() < 1) {
161  // Prepare the image with (semi-)transparent color
162  // Background for colors that are not fully opaque
163  QImage tempBackground = transparencyBackground(devicePixelRatioF());
164  // Paint the color above
165  QPainter(&tempBackground).fillRect(0, 0, size().width(), size().height(), d_pointer->m_color);
166  // Fill a given rectangle with tiles. (QBrush will ignore
167  // the devicePixelRatioF of the image of the tile.)
168  QPainter(&tempImage).fillRect(0, 0, tempImage.width(), tempImage.height(), QBrush(tempBackground));
169  if (layoutDirection() == Qt::RightToLeft) {
170  // Horizontally mirrored image for right-to-left layout,
171  // so that the “nice” part is the first you see in reading
172  // direction.
173  tempImage = tempImage.mirrored(true, // horizontally mirrored
174  false // vertically mirrored
175  );
176  }
177  } else {
178  // Prepare the image with plain color
179  tempImage.fill(d_pointer->m_color);
180  }
181  // Set correct devicePixelRatioF for image
182  tempImage.setDevicePixelRatio(devicePixelRatioF());
183  // Paint the image on the widget
184  widgetPainter.drawImage(contentsMargins().left(), contentsMargins().top(), tempImage);
185 }
