93 QSharedPointer<PerceptualColor::RgbColorSpace> RgbColorSpace::createFromFile(const QString &fileName)
94 {
95  cmsIOHANDLER *myIOHandler = IOHandlerFactory::createReadOnly(nullptr, fileName);
96  if (myIOHandler == nullptr) {
97  return nullptr;
98  }
99 
100  cmsHPROFILE myProfileHandle = cmsOpenProfileFromIOhandlerTHR( //
101  nullptr, // ContextID
102  myIOHandler // IO handler
103  );
104  if (myProfileHandle == nullptr) {
105  // We do not have to delete myIOHandler manually.
106  // (cmsOpenProfileFromIOhandlerTHR did that when
107  // failing to open the profile handle.)
108  return nullptr;
109  }
110 
111  // Create an invalid object:
112  QSharedPointer<PerceptualColor::RgbColorSpace> newObject {new RgbColorSpace()};
113  // Try to transform it into a valid object:
114  const bool success = newObject->d_pointer->initialize(myProfileHandle);
115  // Clean up
116  cmsCloseProfile(myProfileHandle);
117  // We do not have to delete myIOHandler manually.
118  // (myCmsProfileHandle did that when cmsCloseProfile() was invoked.)
119 
120  // Return

123  }

139 bool RgbColorSpace::RgbColorSpacePrivate::initialize(cmsHPROFILE rgbProfileHandle)
140 {
141  m_cmsInfoDescription = getInformationFromProfile(rgbProfileHandle, cmsInfoDescription);
142  m_cmsInfoCopyright = getInformationFromProfile(rgbProfileHandle, cmsInfoCopyright);
143  m_cmsInfoManufacturer = getInformationFromProfile(rgbProfileHandle, cmsInfoManufacturer);
144  m_cmsInfoModel = getInformationFromProfile(rgbProfileHandle, cmsInfoModel);
145 
146  // Create an ICC v4 profile object for the Lab color space.
147  cmsHPROFILE labProfileHandle = cmsCreateLab4Profile(
148  // nullptr means: Default white point (D50)
149  // TODO Does this make sense? sRGB white point is D65!
150  nullptr);
151 
152  // Create the transforms
153  // We use the flag cmsFLAGS_NOCACHE which disables the 1-pixel-cache
154  // which is normally used in the transforms. We do this because transforms
155  // that use the 1-pixel-cache are not thread-save. And disabling it
156  // should not have negative impacts as we usually work with gradients,
157  // so anyway it is not likely to have two consecutive pixels with
158  // the same color, which is the only situation where the 1-pixel-cache
159  // makes processing faster.
160  m_transformLabToRgbHandle = cmsCreateTransform(
161  // Create a transform function and get a handle to this function:
162  labProfileHandle, // input profile handle
163  TYPE_Lab_DBL, // input buffer format
164  rgbProfileHandle, // output profile handle
165  TYPE_RGB_DBL, // output buffer format
166  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
167  cmsFLAGS_NOCACHE // flags
168  );
169  m_transformLabToRgb16Handle = cmsCreateTransform(
170  // Create a transform function and get a handle to this function:
171  labProfileHandle, // input profile handle
172  TYPE_Lab_DBL, // input buffer format
173  rgbProfileHandle, // output profile handle
174  TYPE_RGB_16, // output buffer format
175  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
176  cmsFLAGS_NOCACHE // flags
177  );
178  m_transformRgbToLabHandle = cmsCreateTransform(
179  // Create a transform function and get a handle to this function:
180  rgbProfileHandle, // input profile handle
181  TYPE_RGB_DBL, // input buffer format
182  labProfileHandle, // output profile handle
183  TYPE_Lab_DBL, // output buffer format
184  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
185  cmsFLAGS_NOCACHE // flags
186  );
187  // It is mandatory to close the profiles to prevent memory leaks:
188  cmsCloseProfile(labProfileHandle);
189 
190  // After having closed the profiles, we can now return
191  // (if appropriate) without having memory leaks:
192  if ((m_transformLabToRgbHandle == nullptr) //
193  || (m_transformLabToRgb16Handle == nullptr) //
194  || (m_transformRgbToLabHandle == nullptr) //
195  ) {
196  RgbColorSpacePrivate::deleteTransform(m_transformLabToRgb16Handle);
197  RgbColorSpacePrivate::deleteTransform(m_transformLabToRgbHandle);
198  RgbColorSpacePrivate::deleteTransform(m_transformRgbToLabHandle);
199  return false;
200  }
201 
202  // Maximum chroma:
203  // TODO m_maximumChroma should depend on the actual profile.
204  // m_maximumChroma = LchValues::humanMaximumChroma;
205  // m_maximumChroma = 350;
206 
207  // Now we know for sure that lowerChroma is in-gamut and upperChroma is
208  // out-of-gamut…
209  LchDouble candidate;
210  candidate.l = 0;
211  candidate.c = 0;
212  candidate.h = 0;
213  while (!q_pointer->isInGamut(candidate) && (candidate.l < 100)) {
214  candidate.l += gamutPrecision;
215  }
216  m_blackpointL = candidate.l;
217  candidate.l = 100;
218  while (!q_pointer->isInGamut(candidate) && (candidate.l > 0)) {
219  candidate.l -= gamutPrecision;
220  }
221  m_whitepointL = candidate.l;
222  if (m_whitepointL <= m_blackpointL) {
223  qCritical() << "Unable to find blackpoint and whitepoint on gray axis.";
224  throw 0;
225  }
226 
227  // Dirty hacks:
228  QSharedPointer<PerceptualColor::RgbColorSpace> pointerToMainObject;
229  pointerToMainObject.reset(q_pointer);
230  // TODO Self-reference is always a bad idea…
231  m_nearestNeighborSearchImage = new ChromaLightnessImage(pointerToMainObject);
232  const QSize nearestNeighborImageSize = QSize(
233  // width:
234  qRound(nearestNeighborSearchImageHeight / 100.0 * LchValues::humanMaximumChroma) + 1,
235  // height:
236  nearestNeighborSearchImageHeight);
237  m_nearestNeighborSearchImage->setImageSize(nearestNeighborImageSize);
238  m_nearestNeighborSearchImage->setBackgroundColor(Qt::transparent);
239 
240  return true;
241 }

535 PerceptualColor::LchDouble RgbColorSpace::nearestInGamutColorByAdjustingChroma(const PerceptualColor::LchDouble &color) const
536 {
537  LchDouble result = color;
538  PolarPointF temp(result.c, result.h);
539  result.c = temp.radial();
540  result.h = temp.angleDegree();
541 
542  // Test special case: If we are yet in-gamut…
543  if (isInGamut(result)) {
544  return result;
545  }
546 
547  // Now we know: We are out-of-gamut…
548  LchDouble lowerChroma {result.l, 0, result.h};
549  LchDouble upperChroma {result};
550  LchDouble candidate;
551  if (isInGamut(lowerChroma)) {
552  // Now we know for sure that lowerChroma is in-gamut
553  // and upperChroma is out-of-gamut…
554  candidate = upperChroma;
555  while (upperChroma.c - lowerChroma.c > gamutPrecision) {
556  // Our test candidate is half the way between lowerChroma
557  // and upperChroma:
558  candidate.c = ((lowerChroma.c + upperChroma.c) / 2);
559  if (isInGamut(candidate)) {
560  lowerChroma = candidate;
561  } else {
562  upperChroma = candidate;
563  }
564  }
565  result = lowerChroma;
566  } else {
567  if (result.l < d_pointer->m_blackpointL) {
568  result.l = d_pointer->m_blackpointL;
569  result.c = 0;
570  } else {
571  if (result.l > d_pointer->m_whitepointL) {
572  result.l = d_pointer->m_blackpointL;
573  result.c = 0;
574  }
575  }
576  }
577 
578  return result;
579 }

581 PerceptualColor::LchDouble RgbColorSpace::nearestInGamutColorByAdjustingChromaLightness(const PerceptualColor::LchDouble &color)
582 {
583  // TODO Would be great if this function could be const
584 
585  // Initialization
586  LchDouble temp = color;
587  if (temp.c < 0) {
588  temp.c = 0;
589  }
590 
591  if (isInGamut(temp)) {
592  return temp;
593  }
594 
595  QPoint myPixelPosition( //
596  qRound(temp.c * (d_pointer->nearestNeighborSearchImageHeight - 1) / 100.0),
597  qRound(d_pointer->nearestNeighborSearchImageHeight - 1 - temp.l * (d_pointer->nearestNeighborSearchImageHeight - 1) / 100.0));
598  d_pointer->m_nearestNeighborSearchImage->setHue(temp.h);
599  myPixelPosition = d_pointer->nearestNeighborSearch( //
600  myPixelPosition,
601  d_pointer->m_nearestNeighborSearchImage->getImage());
602  LchDouble result = temp;
603  result.c = myPixelPosition.x() * 100.0 / (d_pointer->nearestNeighborSearchImageHeight - 1);
604  result.l = 100 - myPixelPosition.y() * 100.0 / (d_pointer->nearestNeighborSearchImageHeight - 1);
605  return result;
606 }

635 QPoint RgbColorSpace::RgbColorSpacePrivate::nearestNeighborSearch(const QPoint originalPoint, const QImage &image)
636 {
637  // Test for special case:
638  // originalPoint itself is within the image and non-transparent
639  if (image.valid(originalPoint)) {
640  if (image.pixelColor(originalPoint).alpha() == 255) {
641  return originalPoint;
642  }
643  }
644 
645  // No special case. So we have to actually perform
646  // a nearest-neighbor-search.
647  int x;
648  int y;
649  int currentBestX = 0; // 0 is the fallback value
650  int currentBestY = 0; // 0 is the fallback value
651  int currentBestDistanceSquare = std::numeric_limits<int>::max();
652  int x_distance;
653  int y_distance;
654  int temp;
655  for (x = 0; x < image.width(); x++) {
656  for (y = 0; y < image.height(); y++) {
657  if (image.pixelColor(x, y).alpha() == 255) {
658  x_distance = originalPoint.x() - x;
659  y_distance = originalPoint.y() - y;
660  temp = x_distance * x_distance + y_distance * y_distance;
661  if (temp < currentBestDistanceSquare) {
662  currentBestX = x;
663  currentBestY = y;
664  currentBestDistanceSquare = temp;
665  }
666  }
667  }
668  }
669  return QPoint(currentBestX, currentBestY);
670 }

682 QString RgbColorSpace::RgbColorSpacePrivate::getInformationFromProfile(cmsHPROFILE profileHandle, cmsInfoType infoType)
683 {
684  // Initialize a char array of 3 values (two for actual characters and a
685  // one for a terminating null)cmsFloat64Number
686  // The recommended default value for language
687  // following LittleCMS documentation is “en”.
688  char languageCode[3] = "en";
689  // The recommended default value for country
690  // following LittleCMS documentation is “US”.
691  char countryCode[3] = "US";
692  // Update languageCode and countryCode to the actual locale (if possible)
693  const QStringList list = QLocale().name().split(QStringLiteral(u"_"));
694  // The locale codes should be ASCII only, so QString::toUtf8() should
695  // return ASCII-only valid results. We do not know what character encoding
696  // LittleCMS expects, but ASCII seems a save choise.
697  const QByteArray currentLocaleLanguage = list.at(0).toUtf8();
698  const QByteArray currentLocaleCountry = list.at(1).toUtf8();
699  if (currentLocaleLanguage.size() == 2) {
700  languageCode[0] = currentLocaleLanguage.at(0);
701  languageCode[1] = currentLocaleLanguage.at(1);
702  // No need for languageCode[2] = 0; for null-terminated string,
703  // because the array was yet initialized
704  if (currentLocaleCountry.size() == 2) {
705  countryCode[0] = currentLocaleCountry.at(0);
706  countryCode[1] = currentLocaleCountry.at(1);
707  // No need for countryCode[2] = 0; for null-terminated string,
708  // because the array was yet initialized
709  }
710  }
711  // Calculate the size of the buffer that we have to provide for
712  // cmsGetProfileInfo in order to return a value.
713  const cmsUInt32Number resultLength = cmsGetProfileInfo(
714  // Profile in which we search:
715  profileHandle,
716  // The type of information we search:
717  infoType,
718  // The preferred language in which we want to get the information:
719  languageCode,
720  // The preferred country for which we want to get the information:
721  countryCode,
722  // Do not actually provide the information,
723  // just return the required buffer size:
724  nullptr,
725  // Do not actually provide the information,
726  // just return the required buffer size:
727  0);
728  // For the actual buffer size, increment by 1. This helps us to
729  // guarantee a null-terminated string later on.
730  const cmsUInt32Number bufferLength = resultLength + 1;
731 
732  // Allocate the buffer
733  wchar_t *buffer = new wchar_t[bufferLength];
734  // Initialize the buffer with 0
735  for (cmsUInt32Number i = 0; i < bufferLength; ++i) {
736  *(buffer + i) = 0;
737  }
738 
739  // Write the actual information to the buffer
740  cmsGetProfileInfo(
741  // profile in which we search
742  profileHandle,
743  // the type of information we search
744  infoType,
745  // the preferred language in which we want to get the information
746  languageCode,
747  // the preferred country for which we want to get the information
748  countryCode,
749  // the buffer into which the requested information will be written
750  buffer,
751  // the buffer size as previously calculated
752  resultLength);
753  // Make absolutely sure the buffer is null-terminated by marking its last
754  // element (the one that was the +1 "extra" element) as null.
755  *(buffer + (bufferLength - 1)) = 0;
756 
757  // Create a QString() from the from the buffer
758  // cmsGetProfileInfo returns often strings that are smaller than the
759  // previously calculated buffer size. But it seems they are
760  // null-terminated strings. So we read only up to the first null value.
761  // This is save, because we made sure previously that the buffer is
762  // indeed null-terminated.
763  // QString::fromWCharArray will return a QString. It accepts arrays of
764  // wchar_t. wchar_t might have different sizes, either 16 bit or 32 bit
765  // depending on the operating system. As Qt’s documantation of
766  // QString::fromWCharArray() says:
767  //
768  // “If wchar is 4 bytes, the string is interpreted as UCS-4,
769  // if wchar is 2 bytes it is interpreted as UTF-16.”
770  //
771  // However, apparently this is not exact: When wchar is 4 bytes, surrogate
772  // pairs in the code unit array are interpretated like UTF-16: The
773  // surrogate pair is recognized as such. TODO static_assert that this is
774  // true (which seems complicate: better provide a unit test!) (Which
775  // is not strictly UTF-32 conform). Single surrogates cannot be
776  // interpretated correctly, but there will be no crash:
777  // QString::fromWCharArray will continue to read, also the part after
778  // the first UTF error. We can rely on this behaviour: As we do
779  // not really know the encoding of the buffer that LittleCMS returns.
780  // Therefore, it is a good idea to be compatible for various
781  // interpretations.
782  QString result = QString::fromWCharArray(
783  // Convert to string with these parameters:
784  buffer, // read from this buffer
785  -1 // read until the first null element
786  );
787 
788  // Free allocated memory of the buffer
