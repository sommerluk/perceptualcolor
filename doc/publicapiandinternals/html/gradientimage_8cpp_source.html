58 LchaDouble GradientImage::completlyNormalizedAndBounded(const LchaDouble &color)
59 {
60  LchaDouble result;
61  if (color.c < 0) {
62  result.c = color.c * (-1);
63  result.h = fmod(color.h + 180, 360);
64  } else {
65  result.c = color.c;
66  result.h = fmod(color.h, 360);
67  }
68  if (result.h < 0) {
69  result.h += 360;
70  }
71  result.l = qBound<qreal>(0, color.l, 100);
72  result.a = qBound<qreal>(0, color.a, 1);
73  return result;
74 }

108 void GradientImage::updateSecondColor()
109 {
110  m_secondColorCorrectedAndAltered = completlyNormalizedAndBounded(m_secondColorCorrectedAndAltered);
111  if (qAbs(m_firstColorCorrected.h - m_secondColorCorrectedAndAltered.h) > 180) {
112  if (m_firstColorCorrected.h > m_secondColorCorrectedAndAltered.h) {
113  m_secondColorCorrectedAndAltered.h += 360;
114  } else {
115  m_secondColorCorrectedAndAltered.h -= 360;
116  }
117  }

128 QImage GradientImage::getImage()
129 {
130  // If image is in cache, simply return the cache.
131  if (!m_image.isNull()) {
132  return m_image;
133  }
134 
135  // If no cache is available (m_image.isNull()), render a new image.
136 
137  // Special case: zero-size-image
138  if ((m_gradientLength <= 0) || (m_gradientThickness <= 0)) {
139  return m_image;
140  }
141 
142  // If no image is in cache, create a new one (in the cache) and return it.
143 
144  // First, create an image of the gradient with only one pixel thickness.
145  // (Color management operations are expensive in CPU time; we try to
146  // minimize this.)
147  QImage temp(m_gradientLength, 1, QImage::Format_ARGB32_Premultiplied);
148  temp.fill(Qt::transparent); // Initialize the image with transparency.
149  LchaDouble color;
150  for (int i = 0; i < m_gradientLength; ++i) {
151  color = colorFromValue((i + 0.5) / static_cast<qreal>(m_gradientLength));
152  temp.setPixelColor(i, 0, m_rgbColorSpace->toQColorRgbBound(color));
153  }
154 
155  // Now, create a full image of the gradient
156  m_image = QImage(m_gradientLength, m_gradientThickness, QImage::Format_ARGB32_Premultiplied);
157  QPainter painter(&m_image);
158  // Transparency background
159  if ((m_firstColorCorrected.a != 1) || (m_secondColorCorrectedAndAltered.a != 1)) {
160  // Fill the image with tiles. (QBrush will ignore
161  // the devicePixelRatioF of the image of the tile.)
162  painter.fillRect(0, 0, m_gradientLength, m_gradientThickness, QBrush(transparencyBackground(m_devicePixelRatioF)));
163  }
164  // Paint the gradient itself.
165  for (int i = 0; i < m_gradientThickness; ++i) {
166  painter.drawImage(0, i, temp);
167  }
168 
169  // Set the correct scaling information for the image and return
170  m_image.setDevicePixelRatio(m_devicePixelRatioF);
171  return m_image;
172 }

181 LchaDouble GradientImage::colorFromValue(qreal value) const
182 {
183  LchaDouble color;
184  color.l = m_firstColorCorrected.l + (m_secondColorCorrectedAndAltered.l - m_firstColorCorrected.l) * value;
185  color.c = m_firstColorCorrected.c + (m_secondColorCorrectedAndAltered.c - m_firstColorCorrected.c) * value;
186  color.h = m_firstColorCorrected.h + (m_secondColorCorrectedAndAltered.h - m_firstColorCorrected.h) * value;
187  color.a = m_firstColorCorrected.a + (m_secondColorCorrectedAndAltered.a - m_firstColorCorrected.a) * value;
188  return color;
189 }
