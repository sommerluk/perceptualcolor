54 WheelColorPicker::WheelColorPicker(const QSharedPointer<PerceptualColor::RgbColorSpace> &colorSpace, QWidget *parent)
55  : AbstractDiagram(parent)
56  , d_pointer(new WheelColorPickerPrivate(this))
57 {
58  d_pointer->m_rgbColorSpace = colorSpace;
59  d_pointer->m_colorWheel = new ColorWheel(colorSpace, this);
60  d_pointer->m_chromaLightnessDiagram = new ChromaLightnessDiagram(
61  // Same color space for this widget:
62  colorSpace,
63  // This widget is smaller than the color wheel. It will be a child
64  // of the color wheel, so that missed mouse or key events will be
65  // forwarded to the parent widget (color wheel).
66  d_pointer->m_colorWheel);
67  d_pointer->m_colorWheel->setFocusProxy(d_pointer->m_chromaLightnessDiagram);
68  d_pointer->resizeChildWidgets();
69 
70  connect(
71  // changes on the color wheel trigger a change in the inner diagram
72  d_pointer->m_colorWheel,
73  &ColorWheel::hueChanged,
74  this,
75  [this](const qreal newHue) {
76  LchDouble lch = d_pointer->m_chromaLightnessDiagram->currentColor();
77  lch.h = newHue;
78  // We have to be sure that the color is in-gamut also for the
79  // new hue. If it is not, we adjust it:
80  lch = d_pointer->m_rgbColorSpace->nearestInGamutColorByAdjustingChromaLightness(lch);
81  d_pointer->m_chromaLightnessDiagram->setCurrentColor(lch);
82  });
83  connect(d_pointer->m_chromaLightnessDiagram,
84  &ChromaLightnessDiagram::currentColorChanged,
85  this,
86  // As value is stored anyway within ChromaLightnessDiagram member,
87  // it’s enough to just emit the corresponding signal of this class:
88  &WheelColorPicker::currentColorChanged);
89  connect(
90  // QWidget’s constructor requires a QApplication object. As this
91  // is a class derived from QWidget, calling qApp is save here.
92  qApp,
93  &QApplication::focusChanged,
94  d_pointer.get(), // Without .get() apparently connect() won’t work…
95  &WheelColorPickerPrivate::handleFocusChanged);
96 
97  // Initial color
98  setCurrentColor(
99  // Though LchValues::srgbVersatileInitialColor() is expected to
100  // be in-gamut, its more secure to guarantee this explicitly:
101  d_pointer->m_rgbColorSpace->nearestInGamutColorByAdjustingChromaLightness(
102  // Default sRGB initial color:
103  LchValues::srgbVersatileInitialColor()));
104 }

158 QSizeF WheelColorPicker::WheelColorPickerPrivate::optimalChromaLightnessDiagramSize() const
159 {
179  const qreal r = 100.0 / m_rgbColorSpace->maximumChroma();
180  const qreal h = m_chromaLightnessDiagram->d_pointer->leftBorderPhysical() + m_chromaLightnessDiagram->d_pointer->defaultBorderPhysical();
181  const qreal v = 2 * m_chromaLightnessDiagram->d_pointer->defaultBorderPhysical();
182  const qreal d = m_colorWheel->d_pointer->innerDiameter();
183 
265  const qreal x = (1 + qPow(r, 2)); // x = 1 + r²
266  const qreal a =
267  // The square root:
268  qSqrt(
269  // First fraction:
270  d * d / x
271  // Second fraction:
272  + qPow((h + r * v) / x, 2)
273  // Thierd fraction:
274  - (h * h + v * v) / x)
275  // The part after the square root:
276  - (h + r * v) / x;
277  const qreal b = r * a;
278 
279  return QSizeF(a + h, // width
280  b + v // height
281  );
282 }

292 void WheelColorPicker::WheelColorPickerPrivate::resizeChildWidgets()
293 {
294  // Set new geometry of color wheel. Only the size changes, while the
295  // position (which is 0, 0) remains always unchanged.
296  m_colorWheel->resize(q_pointer->size());
297 
298  // Calculate new size for chroma-lightness-diagram
299  const QSizeF widgetSize = optimalChromaLightnessDiagramSize();
300 
301  // Calculate new top-left corner position for chroma-lightness-diagram
302  // (relative to parent widget)
303  const qreal radius = m_colorWheel->maximumWidgetSquareSize() / 2.0;
304  const QPointF widgetTopLeftPos(
305  // x position
306  radius - widgetSize.width() / 2.0,
307  // y position:
308  radius - widgetSize.height() / 2.0);
309 
310  // Correct the new geometry of chroma-lightness-diagram to fit into
311  // an integer raster.
312  QRectF diagramGeometry(widgetTopLeftPos, widgetSize);
313  // We have to round to full integers, so that our integar-based rectangle
314  // does not exceed the dimensions of the floating-point rectangle.
315  // Round to bigger coordinates for top-left corner:
316  diagramGeometry.setLeft(ceil(diagramGeometry.left()));
317  diagramGeometry.setTop(ceil(diagramGeometry.top()));
318  // Round to smaller coordinates for bottom-right corner:
319  diagramGeometry.setRight(floor(diagramGeometry.right()));
320  diagramGeometry.setBottom(floor(diagramGeometry.bottom()));
321  // TODO The rounding has probably changed the ratio (b ÷ a) of the
322  // diagram itself with the chroma-hue widget. Therefore, maybe a little
323  // bit of gammut is not visible at the right of the diagram. There
324  // might be two possibilities to solve this: Either ChromaLightnessDiagram
325  // gets support for scaling to user-defined maximum chroma (unlikely)
326  // or we implement it here, just by reducing a little bit the height
327  // of the widget until the full gamut gets in (easier).
328 
329  // Apply new geometry
330  m_chromaLightnessDiagram->setGeometry(diagramGeometry.toRect());
331 }

361 QSize WheelColorPicker::minimumSizeHint() const
362 {
363  const QSizeF minimumDiagramSize =
364  // Get the mimimum size of the chroma-lightness widget.
365  d_pointer->m_chromaLightnessDiagram->minimumSizeHint()
366  // We have to fit this in a widget pixel raster. But the perfect
367  // position might be between two integer coordinates. We might
368  // have to shift up to 1 pixel at each of the four margins.
369  + QSize(2, 2);
370  const int diameterForMinimumDiagramSize =
371  // The minimum inner diameter of the color wheel has
372  // to be equal (or a little bit bigger) than the
373  // diagonal through the chroma-lightness widget.
374  qCeil(
375  // c = √(a² + b²)
376  qSqrt(qPow(minimumDiagramSize.width(), 2) + qPow(minimumDiagramSize.height(), 2)))
377  // Add size for the color wheel gradient
378  + d_pointer->m_colorWheel->gradientThickness()
379  // Add size for the border around the color wheel gradient
380  + d_pointer->m_colorWheel->d_pointer->border();
381  // Necessary size for this widget so that the diagram fits:
382  const QSize sizeForMinimumDiagramSize(diameterForMinimumDiagramSize, // x
383  diameterForMinimumDiagramSize // y
384  );
385 
386  return sizeForMinimumDiagramSize
387  // Expand to the minimumSizeHint() of the color wheel itself
388  .expandedTo(d_pointer->m_colorWheel->minimumSizeHint())
389  // Expand to the global minimum size for GUI elements
390  .expandedTo(QApplication::globalStrut());
391 }
