82 QValidator::State ExtendedDoubleValidator::validate(QString &input, int &pos) const
83 {
84  QString myInput = input;
85  int myPos = pos;
86 
87  // IF (m_prefix.isEmpty && !m_prefix.isNull)
88  // THEN input.startsWith(m_prefix)
89  // → will be true IF !input.isEmpty
90  // → will be false IF input.isEmpty
91  // This is inconsistent. Therefore, we test is m_prefix is empty.
92  // If not, we do nothing.
93  // The same also applies to suffix.
94  // TODO Bug report, because Qt documentation says at
95  // https://doc.qt.io/qt-6/qstring.html#distinction-between-null-and-empty-strings
96  // (and at its Qt-5-counterpart):
97  // “All functions except isNull() treat null strings the same
98  // as empty strings.”
99  // This is apparently wrong (at least for Qt 5).
100  if (!d_pointer->m_prefix.isEmpty()) {
101  if (myInput.startsWith(d_pointer->m_prefix)) {
102  myInput.remove(0, d_pointer->m_prefix.size());
103  myPos -= d_pointer->m_prefix.size();
104  } else {
105  return QValidator::State::Invalid;
106  }
107  }
108  if (!d_pointer->m_suffix.isEmpty()) {
109  if (myInput.endsWith(d_pointer->m_suffix)) {
110  myInput.chop(d_pointer->m_suffix.size());
111  } else {
112  return QValidator::State::Invalid;
113  }
114  }
115 
116  QValidator::State result = QDoubleValidator::validate(myInput, myPos);
117  // Following the Qt documentation, QDoubleValidator::validate() is allowed
118  // and intended to make changes the arguments passed by reference. We
119  // have to write back these changes also in this reimplemented function.
120  input = d_pointer->m_prefix + myInput + d_pointer->m_suffix;
121  pos = myPos + d_pointer->m_prefix.size();
122 
123  return result;
124 }
