164 QImage ColorWheelImage::getImage()
165 {
166  // If image is in cache, simply return the cache.
167  if (!m_image.isNull()) {
168  return m_image;
169  }
170 
171  // If no cache is available (m_image.isNull()), render a new image.
172 
173  // Special case: zero-size-image
174  if (m_imageSizePhysical <= 0) {
175  return m_image;
176  }
177 
178  // construct our final QImage with transparent background
179  m_image = QImage(QSize(m_imageSizePhysical, m_imageSizePhysical), QImage::Format_ARGB32_Premultiplied);
180  m_image.fill(Qt::transparent);
181 
182  // Calculate diameter of the outer circle
183  const qreal outerCircleDiameter = m_imageSizePhysical - 2 * m_borderPhysical;
184 
185  // Special case: an empty image
186  if (outerCircleDiameter <= 0) {
187  // Make sure to return a completly transparent image.
188  // If we would continue, in spite of an outer diameter of 0,
189  // we might get a non-transparent pixel in the middle.
190  // Set the correct scaling information for the image and return
191  m_image.setDevicePixelRatio(m_devicePixelRatioF);
192  return m_image;
193  }
194 
195  // Generate a temporary non-anti-aliased, intermediate, color wheel,
196  // but with some pixels extra at the inner and outer side. The overlap
197  // defines an overlap for the wheel, so there are some more pixels that
198  // are drawn at the outer and at the inner border of the wheel, to allow
199  // later clipping with anti-aliasing
200  PolarPointF polarCoordinates;
201  int x;
202  int y;
203  QColor rgbColor;
204  LchDouble lch;
205  qreal center = (m_imageSizePhysical - 1) / static_cast<qreal>(2);
206  m_image = QImage(QSize(m_imageSizePhysical, m_imageSizePhysical), QImage::Format_ARGB32_Premultiplied);
207  // Because there may be out-of-gamut colors for some hue (depending on the
208  // given lightness and chroma value) which are drawn transparent, it is
209  // important to initialize this image with a transparent background.
210  m_image.fill(Qt::transparent);
211  lch.l = LchValues::neutralLightness;
212  lch.c = LchValues::srgbVersatileChroma;
213  // minimumRadial: Adding "+ 1" would reduce the workload (less pixel to
214  // process) and still work mostly, but not completely. It creates sometimes
215  // artifacts in the anti-aliasing process. So we don't do that.
216  const qreal minimumRadial = center - m_wheelThicknessPhysical - m_borderPhysical - overlap;
217  const qreal maximumRadial = center - m_borderPhysical + overlap;
218  for (x = 0; x < m_imageSizePhysical; ++x) {
219  for (y = 0; y < m_imageSizePhysical; ++y) {
220  polarCoordinates = PolarPointF(QPointF(x - center, center - y));
221  if (isInRange<qreal>(minimumRadial, polarCoordinates.radial(), maximumRadial)
222 
223  ) {
224  // We are within the wheel
225  lch.h = polarCoordinates.angleDegree();
226  rgbColor = m_rgbColorSpace->toQColorRgbUnbound(lch);
227  if (rgbColor.isValid()) {
228  m_image.setPixelColor(x, y, rgbColor);
229  }
230  }
231  }
232  }
233 
234  // Anti-aliased cut off everything outside the circle (that
235  // means: the overlap)
236  // The natural way would be to simply draw a circle with
237  // QPainter::CompositionMode_DestinationIn which should make transparent
238  // everything that is not in the circle. Unfourtunally, this does not
239  // seem to work. Therefore, we use a workaround and draw a very think
240  // circle outline around the circle with QPainter::CompositionMode_Clear.
241  const qreal circleRadius = outerCircleDiameter / 2;
242  const qreal cutOffThickness = qSqrt(qPow(m_imageSizePhysical, 2) * 2) / 2 // Â½ of image diagonal
243  - circleRadius // circle radius
244  + overlap; // just to be sure
245  QPainter myPainter(&m_image);
246  myPainter.setRenderHint(QPainter::Antialiasing, true);
247  myPainter.setPen(QPen(Qt::SolidPattern, cutOffThickness));
248  myPainter.setCompositionMode(QPainter::CompositionMode_Clear);
249  myPainter.drawEllipse(QPointF(static_cast<qreal>(m_imageSizePhysical) / 2,
250  static_cast<qreal>(m_imageSizePhysical) / 2), // center
251  circleRadius + cutOffThickness / 2, // width
252  circleRadius + cutOffThickness / 2 // height
253  );
254 
255  // set the inner circle of the wheel to anti-aliased transparency
256  const qreal innerCircleDiameter = m_imageSizePhysical - 2 * (m_wheelThicknessPhysical + m_borderPhysical);
257  if (innerCircleDiameter > 0) {
258  myPainter.setCompositionMode(QPainter::CompositionMode_Clear);
259  myPainter.setRenderHint(QPainter::Antialiasing, true);
260  myPainter.setPen(QPen(Qt::NoPen));
261  myPainter.setBrush(QBrush(Qt::SolidPattern));
262  myPainter.drawEllipse(QRectF(m_wheelThicknessPhysical + m_borderPhysical, m_wheelThicknessPhysical + m_borderPhysical, innerCircleDiameter, innerCircleDiameter));
263  }
264 
265  // Set the correct scaling information for the image and return
266  m_image.setDevicePixelRatio(m_devicePixelRatioF);
267  return m_image;
268 }
