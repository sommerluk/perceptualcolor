117 QImage ChromaLightnessImage::getImage()
118 {
119  // If there is an image in cache, simply return the cache.
120  if (!m_image.isNull()) {
121  return m_image;
122  }
123 
124  // If no image is in cache, create a new one (in the cache) with
125  // correct image size.
126  m_image = QImage(m_imageSizePhysical, QImage::Format_ARGB32_Premultiplied);
127  // Test if image size is empty.
128  if (m_image.size().isEmpty()) {
129  // The image must be non-empty (otherwise, our algorithm would
130  // crash because of a division by 0).
131  return m_image;
132  }
133 
134  // Initialization
135  LchDouble LCh;
136  QColor rgbColor;
137  int x;
138  int y;
139  const int imageHeight = m_imageSizePhysical.height();
140  const int imageWidth = m_imageSizePhysical.width();
141 
142  // Initialize the image background
143  if (m_backgroundColor.isValid()) {
144  m_image.fill(m_backgroundColor);
145  } else {
146  m_image.fill(m_rgbColorSpace->toQColorRgbBound(LchValues::neutralGray()));
147  }
148 
149  // Paint the gamut.
150  LCh.h = PolarPointF::normalizedAngleDegree(m_hue);
151  for (y = 0; y < imageHeight; ++y) {
152  LCh.l = 100 - (y + 0.5) * 100.0 / imageHeight;
153  for (x = 0; x < imageWidth; ++x) {
154  // Using the same scale as on the y axis. floating point
155  // division thanks to 100 which is a "cmsFloat64Number"
156  LCh.c = (x + 0.5) * 100.0 / imageHeight;
157  rgbColor = m_rgbColorSpace->toQColorRgbUnbound(LCh);
158  if (rgbColor.isValid()) {
159  // The pixel is within the gamut
160  m_image.setPixelColor(x, y, rgbColor);
161  // If color is out-of-gamut: We have chroma on the x axis and
162  // lightness on the y axis. We are drawing the pixmap line per
163  // line, so we go for given lightness from low chroma to high
164  // chroma. Because of the nature of most gamuts, if once in a
165  // line we have an out-of-gamut value, all other pixels that
166  // are more at the right will be out-of-gamut also. So we
167  // could optimize our code and break here. But as we are not
168  // sure about this (we do not know the gamut at compile time)
169  // for the moment we do not optimize the code.
170  }
171  }
172  }
173 
174  // Now return the cache.
175  return m_image;
176 }
