143 void ColorDialog::setCurrentColor(const QColor &color)
144 {
145  QColor temp;
146  if (color.isValid()) {
147  // Make sure that the QColor::spec() is QColor::Spec::Rgb.
148  // QColorDialog apparently calls QColor.rgb() within
149  // its setCurrentColor function; this will however round to 8 bit
150  // per channel. We prefer a more exact convertion to RGB:
151  temp = QColor::fromRgbF( //
152  color.redF(),
153  color.greenF(),
154  color.blueF(),
155  color.alphaF());
156  } else {
157  // For invalid colors same behavior as QColorDialog
158  temp = QColor(Qt::black);
159  }
160  d_pointer->setCurrentColorWithAlpha( //
161  MultiColor::fromRgbQColor(d_pointer->m_rgbColorSpace, temp),
162  temp.alphaF());
163 }

229 void ColorDialog::ColorDialogPrivate::setCurrentOpaqueColor(const PerceptualColor::MultiColor &color, QWidget *const ignoreWidget)
230 {
231  if (m_isColorChangeInProgress || (color == m_currentOpaqueColor)) {
232  // Nothing to do!
233  return;
234  }
235 
236  // If we have really some work to do, block recursive calls of this function
237  m_isColorChangeInProgress = true;
238 
239  // Save currentColor() for later comparison
240  // Using currentColor() makes sure correct alpha treatment!
241  QColor oldQColor = q_pointer->currentColor();
242 
243  // Update m_currentOpaqueColor
244  m_currentOpaqueColor = color;
245 
246  // Variables
247  QColor tempRgbQColor = color.toRgbQColor();
248  tempRgbQColor.setAlpha(255);
249  QList<double> valueList;
250 
251  // Update RGB widget
252  if (m_rgbSpinBox != ignoreWidget) {
253  valueList.clear();
254  valueList.append(tempRgbQColor.redF() * 255);
255  valueList.append(tempRgbQColor.greenF() * 255);
256  valueList.append(tempRgbQColor.blueF() * 255);
257  m_rgbSpinBox->setSectionValues(valueList);
258  }
259 
260  // Update HSV widget
261  if (m_hsvSpinBox != ignoreWidget) {
262  valueList.clear();
263  valueList.append(tempRgbQColor.hsvHueF() * 360);
264  valueList.append(tempRgbQColor.hsvSaturationF() * 255);
265  valueList.append(tempRgbQColor.valueF() * 255);
266  m_hsvSpinBox->setSectionValues(valueList);
267  }
268 
269  // Update HLC widget
270  if (m_hlcSpinBox != ignoreWidget) {
271  m_hlcSpinBox->setSectionValues(m_currentOpaqueColor.toHlc());
272  }
273 
274  // Update RGB hex widget
275  if (m_rgbLineEdit != ignoreWidget) {
276  updateRgbHexButBlockSignals();
277  }
278 
279  // Update lightness selector
280  if (m_lchLightnessSelector != ignoreWidget) {
281  m_lchLightnessSelector->setValue(color.toLch().l / static_cast<qreal>(100));
282  }
283 
284  // Update chroma-hue diagram
285  if (m_chromaHueDiagram != ignoreWidget) {
286  m_chromaHueDiagram->setCurrentColor(color.toLch());
287  }
288 
289  // Update wheel color picker
290  if (m_wheelColorPicker != ignoreWidget) {
291  m_wheelColorPicker->setCurrentColor(m_currentOpaqueColor.toLch());
292  }
293 
294  // Update alpha gradient slider
295  if (m_alphaGradientSlider != ignoreWidget) {
296  LchaDouble tempColor;
297  tempColor.l = m_currentOpaqueColor.toLch().l;
298  tempColor.c = m_currentOpaqueColor.toLch().c;
299  tempColor.h = m_currentOpaqueColor.toLch().h;
300  tempColor.a = 0;
301  m_alphaGradientSlider->setFirstColor(tempColor);
302  tempColor.a = 1;
303  m_alphaGradientSlider->setSecondColor(tempColor);
304  }
305 
306  // Update widgets that take alpha information
307  if (m_colorPatch != ignoreWidget) {
308  updateColorPatch();
309  }
310 
311  // Emit signal currentColorChanged() only if necessary
312  if (q_pointer->currentColor() != oldQColor) {
313  Q_EMIT q_pointer->currentColorChanged(q_pointer->currentColor());
314  }
315 
316  // End of this function. Unblock resursive
317  // function calls before returning.
318  m_isColorChangeInProgress = false;
319 }

400 void ColorDialog::ColorDialogPrivate::updateRgbHexButBlockSignals()
401 {
402  QSignalBlocker mySignalBlocker(m_rgbLineEdit);
403 
404  // m_currentOpaqueColor is supposed to be always in-gamut. However,
405  // because of rounding issues, a conversion to an unbounded RGB
406  // color could result in an invalid color. Therefore, we must
407  // use a conversion to a _bounded_ RGB color.
408  const QColor rgbColor = m_currentOpaqueColor.toRgbQColor();
409  // We cannot use QColor.name() directly because this function seems
410  // to use floor() instead of round(), which does not make sense in
411  // our dialog, and it would be inconsistend with the other widgets
412  // of the dialog. Therefore, we have to round explicitly (to integers):
413  // This format string provides a non-localized format!
414  // Format of the numbers:
415  // 1) The number itself
416  // 2) The minimal field width (2 digits)
417  // 3) The base of the number representation (16, hexadecimal)
418  // 4) The fill character (leading zero)
419  const QString hexString = QStringLiteral(u"#%1%2%3")
420  .arg(qRound(rgbColor.redF() * 255), 2, 16, QLatin1Char('0'))
421  .arg(qRound(rgbColor.greenF() * 255), 2, 16, QLatin1Char('0'))
422  .arg(qRound(rgbColor.blueF() * 255), 2, 16, QLatin1Char('0'))
423  .toUpper(); // Convert to upper case
424  m_rgbLineEdit->setText(hexString);
425 }

1000 void ColorDialog::ColorDialogPrivate::applyLayoutDimensions()
1001 {
1002  bool collapsedLayout; // true for small layout, false for large layout.
1003  int effectivelyAvailableScreenWidth;
1004  int widthThreeshold;
1005  switch (m_layoutDimensions) {
1006  case DialogLayoutDimensions::collapsed:
1007  collapsedLayout = true;
1008  break;
1009  case DialogLayoutDimensions::expanded:
1010  collapsedLayout = false;
1011  break;
1012  case DialogLayoutDimensions::screenSizeDependent:
1013  // Note: The following code works correctly on scaled
1014  // devices (high-DPI…).
1015 
1016  // We should not use more than 70% of the screen for a dialog.
1017  // That’s roughly the same as the default maximum sizes for
1018  // a QDialog.
1019  effectivelyAvailableScreenWidth = qRound(QGuiApplication::primaryScreen()->availableSize().width() * 0.7);
1020 
1021  // Now we calculate the space we need for displaying the
1022  // graphical selectors and the numerical selector at their
1023  // preferred size in an expanded layout.
1024  // Start with the size of the graphical selectors.
1025  widthThreeshold = qMax(m_wheelColorPicker->sizeHint().width(), m_lightnessFirstWidget->sizeHint().width());
1026  // Add the size of the numerical selector.
1027  widthThreeshold += m_numericalWidget->sizeHint().width();
1028  // Add some space for margins.
1029  widthThreeshold = qRound(widthThreeshold * 1.2);
1030 
1031  // Now decide between collapsed layout and expanded layout
1032  collapsedLayout = (effectivelyAvailableScreenWidth < widthThreeshold);
1033  break;
1034  default:
1035  throw;
1036  }
1037 
1038  if (collapsedLayout) {
1039  if (m_selectorLayout->indexOf(m_numericalWidget) >= 0) {
1040  // Indeed we have expanded layout and have to switch to
1041  // collapsed layout…
1042  m_tabWidget->addTab(m_numericalWidget, tr("&Numeric"));
1043  // We don’t call m_numericalWidget->show(); because this
1044  // is controlled by the QTabWidget.
1045  // Adopt size of dialog to new layout’s size hint:
1046  q_pointer->adjustSize();
1047  }
1048  } else {
1049  if (m_selectorLayout->indexOf(m_numericalWidget) < 0) {
1050  // Indeed we have collapsed layout and have to switch to
1051  // expanded layout…
1052  m_selectorLayout->addWidget(m_numericalWidget);
1053  // We call show because the widget is hidden by removing it
1054  // from its old parent, and needs to be shown explicitly.
1055  m_numericalWidget->show();
1056  // Adopt size of dialog to new layout’s size hint:
1057  q_pointer->adjustSize();
1058  }
1059  }
1060 }
