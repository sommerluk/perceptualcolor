67 QColor AbstractDiagram::focusIndicatorColor() const
68 {
69  // This approach goes wrong on some styles. For example, the Kvantum
70  // style engine provides a style names “KvFlatRed”. This style uses
71  // red to highlight the background of marked text. The focus indicators
72  // however are drawn in orange. This function returns red in this case,
73  // which is wrong. However, I do not know how to get around this
74  // limitation.
75  return palette().color(QPalette::ColorGroup::Active, QPalette::ColorRole::Highlight);
76 }

106 QSize AbstractDiagram::physicalPixelSize() const
107 {
108  // Assert that static_cast<int> always rounds down.
109  static_assert(static_cast<int>(1.9) == 1);
110  static_assert(static_cast<int>(1.5) == 1);
111  static_assert(static_cast<int>(1.0) == 1);
112  // Multiply the size with the (floating point) scale factor
113  // and than round down (by using static_cast<int>).
114  const int width = static_cast<int>(size().width() * devicePixelRatioF());
115  const int height = static_cast<int>(size().height() * devicePixelRatioF());
116  return QSize(width, height);
117 }

209 int AbstractDiagram::gradientThickness() const
210 {
211  ensurePolished();
212  int result = 0;
213  QStyleOptionSlider styleOption;
214  styleOption.initFrom(this); // Sets also QStyle::State_MouseOver
215  styleOption.orientation = Qt::Horizontal;
216  result = qMax(result, style()->pixelMetric(QStyle::PM_SliderThickness, &styleOption, this));
217  styleOption.orientation = Qt::Vertical;
218  result = qMax(result, style()->pixelMetric(QStyle::PM_SliderThickness, &styleOption, this));
219  result = qMax(result, qRound(handleRadius()));
220  // QApplication::globalStrut() is the minimum.
221  // gradientMinimumLength depends on the following checks!
222  result = qMax(result, QApplication::globalStrut().width());
223  result = qMax(result, QApplication::globalStrut().height());
224  // No supplementary space for ticks is added.
225  return result;
226 }
