65 PolarPointF::PolarPointF(const QPointF cartesianCoordiantes)
66 {
67  m_radial = sqrt(pow(cartesianCoordiantes.x(), 2) + pow(cartesianCoordiantes.y(), 2));
68  if (m_radial == 0) {
69  m_angleDegree = 0;
70  return;
71  }
72  if (cartesianCoordiantes.y() >= 0) {
73  m_angleDegree = qRadiansToDegrees(acos(cartesianCoordiantes.x() / m_radial));
74  } else {
75  // M_PI is defined by QtMath (also on platforms that don’t
76  // support it native)
77  m_angleDegree = qRadiansToDegrees(2 * (M_PI)-acos(cartesianCoordiantes.x() / m_radial));
78  }
79 }
