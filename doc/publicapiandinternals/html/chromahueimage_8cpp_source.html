165 QImage ChromaHueImage::getImage()
166 {
167  // If there is an image in cache, simply return the cache.
168  if (!m_image.isNull()) {
169  return m_image;
170  }
171 
172  // If no image is in cache, create a new one (in the cache) with
173  // correct image size.
174  m_image = QImage(QSize(m_imageSizePhysical, m_imageSizePhysical), QImage::Format_ARGB32_Premultiplied);
175  // Calculate the radius of the circle we want to paint (and which will
176  // finally have the background color, while everything around will be
177  // transparent).
178  const qreal circleRadius = (m_imageSizePhysical - 2 * m_borderPhysical) / 2.;
179  if (circleRadius <= 0) {
180  // The border is too big the and image size too small: The size
181  // of the circle is zero. The image will therefore be transparent.
182  // Initialize the image as completly transparent and return.
183  m_image.fill(Qt::transparent);
184  // Set the correct scaling information for the image and return
185  m_image.setDevicePixelRatio(m_devicePixelRatioF);
186  return m_image;
187  }
188  // If we continue, the circle will at least be visible.
189  // Initialize the hole image background to the background color
190  // of the circle.
191  m_image.fill(m_rgbColorSpace->toQColorRgbBound(LchValues::neutralGray()));
192 
193  // Prepare for gamut painting
194  cmsCIELab lab;
195  lab.L = m_lightness;
196  int x;
197  int y;
198  QColor tempColor;
199  const qreal scaleFactor = static_cast<qreal>(2 * m_chromaRange)
200  // The following line will never be 0 because we have have
201  // tested above that circleRadius is > 0, so this line will
202  // we > 0 also.
203  / (m_imageSizePhysical - 2 * m_borderPhysical);
204 
205  // Paint the gamut.
206  // The pixel at position QPoint(x, y) is the square with the top-left
207  // edge at coordinate point QPoint(x, y) and the botton-right edge at
208  // coordinate point QPoint(x+1, y+1). This pixel is supposed to have
209  // the color from coordinate point QPoint(x+0.5, y+0.5), which is
210  // the middle of this pixel. Therefore, with an offset of 0.5 we can
211  // convert from the pixel position to the point in the middle of the pixel.
212  constexpr qreal pixelOffset = 0.5;
213  // TODO Could this be further optimized? For example not go from zero
214  // up to m_imageSizePhysical, but exclude the border (and add the
215  // tolerance)? Tought anyway the color transform (which is the heavy
216  // work) is only done when within a given diameter, reducing loop runs
217  // itself might also increase performance at least a little bit…
218  for (y = 0; y < m_imageSizePhysical; ++y) {
219  lab.b = m_chromaRange - (y + pixelOffset - m_borderPhysical) * scaleFactor;
220  for (x = 0; x < m_imageSizePhysical; ++x) {
221  lab.a = (x + pixelOffset - m_borderPhysical) * scaleFactor - m_chromaRange;
222  if ((qPow(lab.a, 2) + qPow(lab.b, 2)) <= (qPow(m_chromaRange + overlap, 2))) {
223  tempColor = m_rgbColorSpace->toQColorRgbUnbound(lab);
224  if (tempColor.isValid()) {
225  // The pixel is within the gamut!
226  m_image.setPixelColor(x, y, tempColor);
227  }
228  }
229  }
230  }
231 
232  // Cut off everything outside the circle.
233  // If the gamut does not touch the outline of the circle, than
234  // we could also paint directly on m_image, which would save memory.
235  // And the antialiasing result would be the same. If the gamut however
236  // touchs the outline, the antialiasing would be ugly there, therefore
237  // we cut everything outside the circle explicitly.
238  // The natural way would be to simply draw a circle with
239  // QPainter::CompositionMode_DestinationIn which should make transparent
240  // everything that is not in the circle. Unfourtunally, this does not
241  // seem to work. Therefore, we use a workaround and draw a very think
242  // circle outline around the circle with QPainter::CompositionMode_Clear.
243  const qreal cutOffThickness = qSqrt(qPow(m_imageSizePhysical, 2) * 2) / 2 // ½ of image diagonal
244  - circleRadius // circle radius
245  + overlap; // just to be sure
246  QPainter myPainter(&m_image);
247  myPainter.setRenderHint(QPainter::Antialiasing, true);
248  myPainter.setPen(QPen(Qt::SolidPattern, cutOffThickness));
249  myPainter.setCompositionMode(QPainter::CompositionMode_Clear);
250  myPainter.drawEllipse(QPointF(static_cast<qreal>(m_imageSizePhysical) / 2,
251  static_cast<qreal>(m_imageSizePhysical) / 2), // center
252  circleRadius + cutOffThickness / 2, // width
253  circleRadius + cutOffThickness / 2 // height
254  );
255 
256  // Set the correct scaling information for the image and return
257  m_image.setDevicePixelRatio(m_devicePixelRatioF);
258  return m_image;
259 }
