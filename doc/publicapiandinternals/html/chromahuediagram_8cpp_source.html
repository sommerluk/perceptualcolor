52 ChromaHueDiagram::ChromaHueDiagram(const QSharedPointer<PerceptualColor::RgbColorSpace> &colorSpace, QWidget *parent)
53  : AbstractDiagram(parent)
54  , d_pointer(new ChromaHueDiagramPrivate(this, colorSpace))
55 {
56  // Setup LittleCMS. This is the first thing to do, because other
57  // operations rely on a working LittleCMS.
58  d_pointer->m_rgbColorSpace = colorSpace;
59 
60  // Set focus policy
61  // In Qt, usually focus (QWidget::hasFocus()) by mouse click is either
62  // not accepted at all or accepted always for the hole rectangular
63  // widget, depending on QWidget::focusPolicy(). This is not convenient
64  // and intuitive for big, circular-shaped widgets like this one. It
65  // would be nicer if the focus would only be accepted by mouse clicks
66  // <em>within the circle itself</em>. Qt does not provide a build-in
67  // way to do this. But a workaround to implement this behavior is
68  // possible: Set QWidget::focusPolicy() to <em>not</em> accept focus
69  // by mouse click. Then, reimplement mousePressEvent() and call
70  // setFocus(Qt::MouseFocusReason) if the mouse click is within the
71  // circle. Therefore, this class simply defaults to
72  // Qt::FocusPolicy::TabFocus for QWidget::focusPolicy().
73  setFocusPolicy(Qt::FocusPolicy::TabFocus);
74 
75  // Initialize the color
76  setCurrentColor(LchValues::srgbVersatileInitialColor());
77 }

114 void ChromaHueDiagram::mousePressEvent(QMouseEvent *event)
115 {
116  // TODO Also accept out-of-gamut clicks when they are covered by the
117  // current handle.
118  if (d_pointer->isWidgetPixelPositionWithinMouseSensibleCircle(event->pos())) {
119  event->accept();
120  // Mouse focus is handled manually because so we can accept
121  // focus only on mouse clicks within the displayed gamut, while
122  // rejecting focus otherwise. In the constructor, therefore
123  // Qt::FocusPolicy::TabFocus is specified, so that manual handling
124  // of mouse focus is up to this code here.
125  setFocus(Qt::MouseFocusReason);
126  // Enable mouse tracking from now on:
127  d_pointer->m_isMouseEventActive = true;
128  // As clicks are only accepted within the visible gamut, the mouse
129  // cursor is made invisible. Its function is taken over by the
130  // handle itself within the displayed gamut.
131  setCursor(Qt::BlankCursor);
132  // Set the color property
133  d_pointer->setColorFromWidgetPixelPosition(event->pos());
134  // Schedule a paint event, so that the wheel handle will show. It’s
135  // not enough to hope setColorFromWidgetCoordinates() would do this,
136  // because setColorFromWidgetCoordinates() would not update the
137  // widget if the mouse click was done at the same position as the
138  // current color handle.
139  update();
140  } else {
141  // Make sure default behavior like drag-window in KDE’s
142  // “Breeze” widget style works if this widget does not
143  // actually react itself on a mouse event.
144  event->ignore();
145  }
146 }

235 void ChromaHueDiagram::wheelEvent(QWheelEvent *event)
236 {
237  if (
238  // Do nothing while a the mouse is clicked and the mouse movement is
239  // tracked anyway because this would be confusing for the user.
240  (!d_pointer->m_isMouseEventActive) &&
241  // Only react on good old vertical wheels,
242  // and not on horizontal wheels.
243  (event->angleDelta().y() != 0) &&
244  // Only react on wheel events when then happen in the appropriate
245  // area.a
246  d_pointer->isWidgetPixelPositionWithinMouseSensibleCircle(event->pos())
247  // then:
248  ) {
249  event->accept();
250  // Calculate the new hue.
251  // This may result in a hue smaller then 0° or bigger then 360°.
252  // This should not make any problems.
253  LchDouble newColor = d_pointer->m_currentColor;
254  newColor.h += standardWheelStepCount(event) * singleStepHue;
255  setCurrentColor(d_pointer->m_rgbColorSpace->nearestInGamutColorByAdjustingChroma(newColor));
256  } else {
257  event->ignore();
258  }
259 }

292 void ChromaHueDiagram::keyPressEvent(QKeyEvent *event)
293 {
294  LchDouble newColor = currentColor();
295  switch (event->key()) {
296  case Qt::Key_Up:
297  newColor.c += singleStepChroma;
298  break;
299  case Qt::Key_Down:
300  newColor.c -= singleStepChroma;
301  break;
302  case Qt::Key_Left:
303  newColor.h += singleStepHue;
304  break;
305  case Qt::Key_Right:
306  newColor.h -= singleStepHue;
307  break;
308  case Qt::Key_PageUp:
309  newColor.c += pageStepChroma;
310  break;
311  case Qt::Key_PageDown:
312  newColor.c -= pageStepChroma;
313  break;
314  case Qt::Key_Home:
315  newColor.h += pageStepHue;
316  break;
317  case Qt::Key_End:
318  newColor.h -= pageStepHue;
319  break;
320  default:
321  // Quote from Qt documentation:
322  //
323  // “If you reimplement this handler, it is very important that
324  // you call the base class implementation if you do not act
325  // upon the key.
326  //
327  // The default implementation closes popup widgets if the
328  // user presses the key sequence for QKeySequence::Cancel
329  // (typically the Escape key). Otherwise the event is
330  // ignored, so that the widget’s parent can interpret it.“
331  QWidget::keyPressEvent(event);
332  return;
333  }
334  // Here we reach only if the key has been recognized. If not, in the
335  // default branch of the switch statement, we would have passed the
336  // keyPressEvent yet to the parent and returned.
337  if (newColor.c < 0) {
338  // Do not allow negative chroma values.
339  // (Doing so would be counter-intuitive.)
340  newColor.c = 0;
341  }
342  // Move the value into gamut (if necessary):
343  newColor = d_pointer->m_rgbColorSpace->nearestInGamutColorByAdjustingChroma(newColor);
344  // Apply the new value:
345  setCurrentColor(newColor);
346 }

481 cmsCIELab ChromaHueDiagram::ChromaHueDiagramPrivate::fromWidgetPixelPositionToLab(const QPoint position) const
482 {
483  const qreal scaleFactor = (2.0 * m_rgbColorSpace->maximumChroma()) / (q_pointer->maximumWidgetSquareSize() - 2.0 * diagramBorder());
484  // The pixel at position 0 0 has its top left border at position 0 0
485  // and its bottom right border at position 1 1 and its center at
486  // position 0.5 0.5. Its the center of the pixel that is our reference
487  // for conversion, therefore we have to ship by 0.5 widget pixels.
488  constexpr qreal pixelValueShift = 0.5;
489  cmsCIELab lab;
490  lab.L = m_currentColor.l;
491  lab.a = (position.x() + pixelValueShift - diagramOffset()) * scaleFactor;
492  lab.b = (position.y() + pixelValueShift - diagramOffset()) * scaleFactor * (-1);
493  return lab;
494 }

540 bool ChromaHueDiagram::ChromaHueDiagramPrivate::isWidgetPixelPositionWithinMouseSensibleCircle(const QPoint position) const
541 {
542  const qreal radial = PolarPointF(
543  // Position relative to polar coordinate system center:
544  position -
545  diagramCenter()
546  // Apply the offset between
547  // - a pixel position on one hand and
548  // - a coordinat point in the middle of this very
549  // same pixel on the other:
550  + QPointF(0.5, 0.5))
551  .radial();
552 
553  const qreal diagramCircleRadius = q_pointer->maximumWidgetSquareSize() / static_cast<qreal>(2) - diagramBorder();
554 
555  return (radial <= diagramCircleRadius);
556 }

585 void ChromaHueDiagram::paintEvent(QPaintEvent *event)
586 {
587  Q_UNUSED(event);
588 
589  // We do not paint directly on the widget, but on a QImage buffer first:
590  // Render anti-aliased looks better. But as Qt documentation says:
591  //
592  // “Renderhints are used to specify flags to QPainter that may or
593  // may not be respected by any given engine.”
594  //
595  // Painting here directly on the widget might lead to different
596  // anti-aliasing results depending on the underlying window system. This
597  // is especially problematic as anti-aliasing might shift or not a pixel
598  // to the left or to the right. So we paint on a QImage first. As QImage
599  // (at difference to QPixmap and a QWidget) is independent of native
600  // platform rendering, it guarantees identical anti-aliasing results on
601  // all platforms. Here the quote from QPainter class documentation:
602  //
603  // “To get the optimal rendering result using QPainter, you should
604  // use the platform independent QImage as paint device; i.e. using
605  // QImage will ensure that the result has an identical pixel
606  // representation on any platform.”
607  QImage buffer(maximumPhysicalSquareSize(), // width
608  maximumPhysicalSquareSize(), // height
609  QImage::Format_ARGB32_Premultiplied // format
610  );
611  buffer.fill(Qt::transparent);
612  buffer.setDevicePixelRatio(devicePixelRatioF());
613 
614  // Other initialization
615  QPainter bufferPainter(&buffer);
616  QPen pen;
617  const QBrush transparentBrush {Qt::transparent};
618  // Set color of the handle: Black or white, depending on the lightness of
619  // the currently selected color.
620  const QColor handleColor {handleColorFromBackgroundLightness(d_pointer->m_currentColor.l)};
621  const QPointF widgetCoordinatesFromCurrentColor {d_pointer->widgetCoordinatesFromCurrentColor()};
622 
623  // Paint the gamut itself as available in the cache.
624  bufferPainter.setRenderHint(QPainter::Antialiasing, false);
625  // As devicePixelRatioF() might have changed, we make sure everything
626  // that might depend on devicePixelRatioF() is updated before painting.
627  d_pointer->m_chromaHueImage.setBorder(d_pointer->diagramBorder() * devicePixelRatioF());
628  d_pointer->m_chromaHueImage.setImageSize(maximumPhysicalSquareSize());
629  d_pointer->m_chromaHueImage.setChromaRange(d_pointer->m_rgbColorSpace->maximumChroma());
630  d_pointer->m_chromaHueImage.setLightness(d_pointer->m_currentColor.l);
631  d_pointer->m_chromaHueImage.setDevicePixelRatioF(devicePixelRatioF());
632  bufferPainter.drawImage(QPoint(0, 0), // position of the image
633  d_pointer->m_chromaHueImage.getImage() // image
634  );
635 
636  // Paint a color wheel around
637  bufferPainter.setRenderHint(QPainter::Antialiasing, false);
638  // As devicePixelRatioF() might have changed, we make sure everything
639  // that might depend on devicePixelRatioF() is updated before painting.
640  d_pointer->m_wheelImage.setBorder(spaceForFocusIndicator() * devicePixelRatioF());
641  d_pointer->m_wheelImage.setDevicePixelRatioF(devicePixelRatioF());
642  d_pointer->m_wheelImage.setImageSize(maximumPhysicalSquareSize());
643  d_pointer->m_wheelImage.setWheelThickness(gradientThickness() * devicePixelRatioF());
644  bufferPainter.drawImage(QPoint(0, 0), // position of the image
645  d_pointer->m_wheelImage.getImage() // the image itself
646  );
647 
648  // Paint a handle on the color wheel (only if a mouse event is
649  // currently active).
650  if (d_pointer->m_isMouseEventActive) {
651  // The radius of the outer border of the color wheel
652  const qreal radius = maximumWidgetSquareSize() / static_cast<qreal>(2) - spaceForFocusIndicator();
653  // Get widget coordinate point for the handle
654  QPointF myHandleInner = PolarPointF(radius - gradientThickness(), d_pointer->m_currentColor.h).toCartesian();
655  myHandleInner.ry() *= -1; // Transform to Widget coordinate points
656  myHandleInner += d_pointer->diagramCenter();
657  QPointF myHandleOuter = PolarPointF(radius, d_pointer->m_currentColor.h).toCartesian();
658  myHandleOuter.ry() *= -1; // Transform to Widget coordinate points
659  myHandleOuter += d_pointer->diagramCenter();
660  // Draw the line
661  pen = QPen();
662  pen.setWidth(handleOutlineThickness());
663  // TODO Instead of Qt::FlatCap, we could really paint a handle
664  // that does match perfectly the round inner and outer border
665  // of the wheel. But: Is it really worth the complexity?
666  pen.setCapStyle(Qt::FlatCap);
667  pen.setColor(handleColor);
668  bufferPainter.setPen(pen);
669  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
670  bufferPainter.drawLine(myHandleInner, myHandleOuter);
671  }
672 
673  // Paint the handle within the gamut
674  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
675  pen = QPen();
676  pen.setWidth(handleOutlineThickness());
677  pen.setColor(handleColor);
678  pen.setCapStyle(Qt::RoundCap);
679  bufferPainter.setPen(pen);
680  bufferPainter.setBrush(transparentBrush);
681  bufferPainter.drawEllipse(widgetCoordinatesFromCurrentColor, // center
682  handleRadius(), // x radius
683  handleRadius() // y radius
684  );
685  const QPointF diagramCartesianCoordinatesFromCurrentColor {widgetCoordinatesFromCurrentColor.x() - d_pointer->diagramOffset(), (widgetCoordinatesFromCurrentColor.y() - d_pointer->diagramOffset()) * (-1)};
686  PolarPointF diagramPolarCoordinatesFromCurrentColor {diagramCartesianCoordinatesFromCurrentColor};
687  // lineRadial will be a point at the middle of the line thickness
688  // of the circular handle.
689  qreal lineRadial = diagramPolarCoordinatesFromCurrentColor.radial() - handleRadius();
690  if (lineRadial > 0) {
691  QPointF lineEndWidgetCoordinates = PolarPointF(lineRadial, diagramPolarCoordinatesFromCurrentColor.angleDegree()).toCartesian();
692  lineEndWidgetCoordinates.ry() *= (-1);
693  lineEndWidgetCoordinates += d_pointer->diagramCenter();
694  bufferPainter.drawLine(
695  // point 1 (center of the diagram):
696  d_pointer->diagramCenter(),
697  // point 2:
698  lineEndWidgetCoordinates);
699  }
700 
701  // Paint a focus indicator.
702  //
703  // We could paint a focus indicator (round or rectangular) around the
704  // handle. Depending on the currently selected hue for the diagram, it
705  // looks ugly because the colors of focus indicator and diagram do not
706  // harmonize, or it is mostly invisible if the colors are similar. So
707  // this approach does not work well.
708  //
709  // It seems better to paint a focus indicator for the whole widget.
710  // We could use the style primitives to paint a rectangular focus
711  // indicator around the whole widget:
712  //
713  // style()->drawPrimitive(
714  // QStyle::PE_FrameFocusRect,
715  // &option,
716  // &painter,
717  // this
718  // );
719  //
720  // However, this does not work well because this widget does not have a
721  // rectangular form.
722  //
723  // Then we have to design the line that we want to display. It is better
724  // to do that ourselves instead of relying on generic QStyle::PE_Frame or
725  // similar solutions as their result seems to be quite unpredictable
726  // across various styles. So we use handleOutlineThickness as line width
727  // and paint it at the left-most possible position. As m_wheelBorder
728  // accommodates also to handleRadius(), the distance of the focus line to
729  // the real widget also does, which looks nice.
730  if (hasFocus()) {
731  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
732  pen = QPen();
733  pen.setWidth(handleOutlineThickness());
734  pen.setColor(focusIndicatorColor());
735  bufferPainter.setPen(pen);
736  bufferPainter.setBrush(transparentBrush);
737  bufferPainter.drawEllipse(
738  // center:
739  d_pointer->diagramCenter(),
740  // x radius:
741  d_pointer->diagramOffset() - handleOutlineThickness() / 2.0,
742  // y radius:
743  d_pointer->diagramOffset() - handleOutlineThickness() / 2.0);
744  }
745 
746  // Paint the buffer to the actual widget
747  QPainter widgetPainter(this);
748  widgetPainter.setRenderHint(QPainter::Antialiasing, false);
749  widgetPainter.drawImage(QPoint(0, 0), buffer);
750 }
