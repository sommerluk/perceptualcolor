157 QSize MultiSpinBox::sizeHint() const
158 {
159  // This function intentionally does not cache the text string.
160  // Which variant is the longest text string, that depends on the current
161  // font policy. This might have changed since the last call. Therefore,
162  // each time this functin is called, we calculate again the longest
163  // test string (“completeString”).
164 
165  ensurePolished();
166 
167  const QFontMetrics myFontMetrics(fontMetrics());
168  QList<MultiSpinBoxSectionConfiguration> myConfiguration = d_pointer->m_sectionConfigurations;
169  int height = lineEdit()->sizeHint().height();
170  int width = 0;
171  QString textOfMinimumValue;
172  QString textOfMaximumValue;
173  QString completeString;
174 
175  // Get the text for all the sections
176  for (int i = 0; i < myConfiguration.count(); ++i) {
177  // Prefix
178  completeString += myConfiguration.at(i).prefix();
179  // For each section, test if the minimum value or the maximum
180  // takes more space (width). Choose the one that takes more place
181  // (width).
182  textOfMinimumValue = locale().toString( //
183  myConfiguration.at(i).minimum(), // value
184  'f', // format
185  myConfiguration.at(i).decimals() // precision
186  );
187  textOfMaximumValue = locale().toString( //
188  myConfiguration.at(i).maximum(), // value
189  'f', // format
190  myConfiguration.at(i).decimals() // precision
191  );
192  if (myFontMetrics.horizontalAdvance(textOfMinimumValue) > myFontMetrics.horizontalAdvance(textOfMaximumValue)) {
193  completeString += textOfMinimumValue;
194  } else {
195  completeString += textOfMaximumValue;
196  }
197  // Suffix
198  completeString += myConfiguration.at(i).suffix();
199  }
200 
201  // Add some extra space, just as QSpinBox seems to do also.
202  completeString += QStringLiteral(u" ");
203 
204  // Calculate string width and add two extra pixel for cursor
205  // blinking space.
206  width = myFontMetrics.horizontalAdvance(completeString) + 2;
207 
208  // Calculate the final size in pixel
209  QStyleOptionSpinBox myStyleOptionsForSpinBoxes;
210  initStyleOption(&myStyleOptionsForSpinBoxes);
211  QSize contentSize(width, height);
212  // Calculate widget size necessary to display a given content
213  QSize result = style()
214  ->sizeFromContents(QStyle::CT_SpinBox, // type
215  &myStyleOptionsForSpinBoxes, // style options
216  contentSize, // size of the content
217  this // optional widget argument (for better caluclations)
218  )
219  .expandedTo(QApplication::globalStrut());
220 
221  if (d_pointer->m_actionButtonCount > 0) {
222  // Determine the size of icons for actions similar to what Qt
223  // does in QLineEditPrivate::sideWidgetParameters() and than
224  // add this to the size hint.
225  //
226  // This code leads in general to good results. However, some
227  // styles like CDE and Motif calculate badly the size for
228  // QAbstractSpinBox and therefore also for this widget.
229  //
230  // The Kvantum style has a strange behaviour: While behaving
231  // correctly on QSpinBox, it does not behave correctly on small
232  // widget sizes: On small widget sizes, style()->sizeFromContents()
233  // returns a value that is too small, but the actual size of
234  // the widget in the layout system will be bigger than the
235  // reported size, so this looks okay. On big sizes however,
236  // the actual size in the layout system is identical to the
237  // reported size, which is too small. For these styles,
238  // there seems to be no good work-around. These very same
239  // styles also have problems with normal QSpinBox widgets and
240  // cut some of the text there, so I suppose this is rather a
241  // problem of these styles, and not of our code.
242  //
243  // TODO That‘s wrong. Normal spinboxes work fine in all CDE,
244  // Motif and Kvantum styles. It should work also for MultiSpinBox!
245  const int actionButtonIconSize = style()->pixelMetric(QStyle::PM_SmallIconSize, // pixel metric type
246  nullptr, // style options
247  lineEdit() // widget (optional)
248  );
249  const int actionButtonMargin = actionButtonIconSize / 4;
250  const int actionButtonWidth = actionButtonIconSize + 6;
251  // Only 1 margin per button:
252  const int actionButtonSpace = actionButtonWidth + actionButtonMargin;
253  result.setWidth(result.width() + d_pointer->m_actionButtonCount * actionButtonSpace);
254  }
255 
256  return result;
257 }

404 void MultiSpinBox::MultiSpinBoxPrivate::setCurrentIndexWithoutUpdatingText(int newIndex)
405 {
406  if (!isInRange(0, newIndex, m_sectionConfigurations.count() - 1)) {
407  qWarning() << "The function" << __func__ //
408  << "in file" << __FILE__ //
409  << "near to line" << __LINE__ //
410  << "was called with an invalid “newIndex“ argument of" << newIndex //
411  << "thought the valid range is currently [" << 0 << ", " << m_sectionConfigurations.count() - 1 << "]. This is a bug.";
412  throw 0;
413  }
414 
415  if (newIndex == m_currentIndex) {
416  // There is nothing to do here.
417  return;
418  }
419 
420  // Apply the changes
421  m_currentIndex = newIndex;
422  updatePrefixValueSuffixText();
423  m_validator->setPrefix(m_textBeforeCurrentValue);
424  m_validator->setSuffix(m_textAfterCurrentValue);
425  m_validator->setRange(
426  // Minimum:
427  m_sectionConfigurations.at(m_currentIndex).minimum(),
428  // Maximum:
429  m_sectionConfigurations.at(m_currentIndex).maximum());
430 
431  // The state (enabled/disabled) of the buttons “Step up” and “Step down”
432  // has to be updated. To force this, update() is called manually here:
433  q_pointer->update();
434 }

442 QAbstractSpinBox::StepEnabled MultiSpinBox::stepEnabled() const
443 {
444  const MultiSpinBoxSectionConfiguration currentSectionConfiguration = d_pointer->m_sectionConfigurations.at(d_pointer->m_currentIndex);
445  const double currentSectionValue = sectionValues().at(d_pointer->m_currentIndex);
446 
447  // When wrapping is enabled, step up and step down are always possible.
448  if (currentSectionConfiguration.isWrapping()) {
449  return QAbstractSpinBox::StepEnabled(StepUpEnabled | StepDownEnabled);
450  }
451 
452  // When wrapping is not enabled, we have to compare the value with
453  // maximum and minimum.
454  QAbstractSpinBox::StepEnabled result;
455  // Test is step up should be enabled…
456  if (currentSectionValue < currentSectionConfiguration.maximum()) {
457  result.setFlag(StepUpEnabled, true);
458  }
459 
460  // Test is step down should be enabled…
461  if (currentSectionValue > currentSectionConfiguration.minimum()) {
462  result.setFlag(StepDownEnabled, true);
463  }
464  return result;
465 }

536 void MultiSpinBox::MultiSpinBoxPrivate::setSectionValuesWithoutFurtherUpdating(const QList<double> &newSectionValues)
537 {
538  if (newSectionValues.count() < 1) {
539  return;
540  }
541 
542  const int sectionCount = m_sectionConfigurations.count();
543 
544  QList<double> fixedNewSectionValues = newSectionValues;
545 
546  // Adapt the count of values:
547  while (fixedNewSectionValues.count() < sectionCount) {
548  // Add elements if there are not enough:
549  fixedNewSectionValues.append(MultiSpinBoxPrivate::defaultSectionValue);
550  }
551  while (fixedNewSectionValues.count() > sectionCount) {
552  // Remove elements if there are too many:
553  fixedNewSectionValues.removeLast();
554  }
555 
556  // Make sure the new section values are
557  // valid (minimum <= value <= maximum):
558  MultiSpinBoxSectionConfiguration myConfig;
559  double rangeWidth;
560  qreal temp; // TODO Why qreal and not double?
561  for (int i = 0; i < sectionCount; ++i) {
562  myConfig = m_sectionConfigurations.at(i);
563  fixedNewSectionValues[i] =
564  // Round value _before_ applying boundaries/wrapping.
565  roundToDigits(fixedNewSectionValues.at(i), myConfig.decimals());
566  if (myConfig.isWrapping()) {
567  rangeWidth = myConfig.maximum() - myConfig.minimum();
568  if (rangeWidth <= 0) {
569  // This is a speciel case.
570  // This happens when minimum == maximum (or
571  // if minimum > maximum, which is invalid).
572  fixedNewSectionValues[i] = myConfig.minimum();
573  } else {
574  // floating-point modulo (fmod) operation
575  temp = fmod(
576  // Dividend:
577  fixedNewSectionValues.at(i) - myConfig.minimum(),
578  // Divisor:
579  rangeWidth);
580  if (temp < 0) {
581  // Negative results shall be convertet
582  // in positive results:
583  temp += rangeWidth;
584  }
585  temp += myConfig.minimum();
586  fixedNewSectionValues[i] = temp;
587  }
588  } else {
589  fixedNewSectionValues[i] = qBound(
590  // If there is no wrapping, simply bound:
591  myConfig.minimum(),
592  fixedNewSectionValues.at(i),
593  myConfig.maximum());
594  }
595  }
596 
597  if (m_sectionValues != fixedNewSectionValues) {
598  m_sectionValues = fixedNewSectionValues;
599  Q_EMIT q_pointer->sectionValuesChanged(fixedNewSectionValues);
600  }
601 }

798 void MultiSpinBox::MultiSpinBoxPrivate::updateCurrentValueFromText(const QString &lineEditText)
799 {
800  // Get the clean test. That means, we start with “text”, but
801  // we remove the m_currentSectionTextBeforeValue and the
802  // m_currentSectionTextAfterValue, so that only the text of
803  // the value itself remains.
804  QString cleanText = lineEditText;
805  if (cleanText.startsWith(m_textBeforeCurrentValue)) {
806  cleanText.remove(0, m_textBeforeCurrentValue.count());
807  } else {
808  // The text does not start with the correct characters.
809  // This is an error.
810  qWarning() << "The function" << __func__ //
811  << "in file" << __FILE__ //
812  << "near to line" << __LINE__ //
813  << "was called with the invalid “lineEditText“ argument “" << lineEditText //
814  << "” that does not start with the expected character sequence “" << m_textBeforeCurrentValue << ". " //
815  << "The call is ignored. This is a bug.";
816  return;
817  }
818  if (cleanText.endsWith(m_textAfterCurrentValue)) {
819  cleanText.chop(m_textAfterCurrentValue.count());
820  } else {
821  // The text does not start with the correct characters.
822  // This is an error.
823  qWarning() << "The function" << __func__ //
824  << "in file" << __FILE__ //
825  << "near to line" << __LINE__ //
826  << "was called with the invalid “lineEditText“ argument “" << lineEditText //
827  << "” that does not end with the expected character sequence “" << m_textAfterCurrentValue << ". " //
828  << "The call is ignored. This is a bug.";
829  return;
830  }
831 
832  // Update…
833  bool ok;
834  QList<double> myValues = q_pointer->sectionValues();
835  myValues[m_currentIndex] = q_pointer->locale().toDouble(cleanText, &ok);
836  setSectionValuesWithoutFurtherUpdating(myValues);
837  // Make sure that the buttons for step up and step down are updated.
838  q_pointer->update();
839  // The lineEdit()->text() property is intentionally not updated because
840  // this function is ment to receive signals of the very same lineEdit().
841 }

876 void MultiSpinBox::MultiSpinBoxPrivate::reactOnCursorPositionChange(const int oldPos, const int newPos)
877 {
878  // This slot is meant to be connected to the
879  // QLineEdit::cursorPositionChanged() signal of
880  // the MultiSpinBox::lineEdit() child widget.
881  // This signal emits the two “int” parameters “oldPos”
882  // and “newPos”. We only need the second one, but we have
883  // to take both of them as parameter if we want to stay
884  // compatible. Therefore, we mark the first one
885  // with Q_UNUSED to avoid compiler warnings.
886  Q_UNUSED(oldPos);
887 
888  // We are working here with QString::length() and
889  // QLineEdit::cursorPosition(). Both are of type “int”, and both are
890  // measured in UTF-16 code units. While it feels quite uncomfortable
891  // to measure cursor positions in code _units_ and not at least in
892  // in code _points_, it does not matter for this code, as the behaviour
893  // is consistent between both usages.
894 
895  if (isCursorPositionAtCurrentSectionValue(newPos)) {
896  // We are within the value text of our current section value.
897  // There is nothing to do here.
898  return;
899  }
900 
901  QSignalBlocker myBlocker(q_pointer->lineEdit());
902 
903  // The new position is not at the current value, but the old one might
904  // have been. So maybe we have to correct the value, which might change
905  // its length. If the new cursor position is after this value, it will
906  // have to be adapted (if the value had been changed or alternated).
907  const int oldTextLength = q_pointer->lineEdit()->text().length();
908  const bool cursorPositionHasToBeAdaptedToTextLenghtChange = (newPos > (oldTextLength - m_textAfterCurrentValue.length()));
909 
910  // Calculat in which section the cursor is
911  int sectionOfTheNewCursorPosition;
912  int reference = 0;
913  for (sectionOfTheNewCursorPosition = 0; sectionOfTheNewCursorPosition < m_sectionConfigurations.count() - 1; ++sectionOfTheNewCursorPosition) {
914  reference += m_sectionConfigurations.at(sectionOfTheNewCursorPosition).prefix().length();
915  reference += formattedValue(sectionOfTheNewCursorPosition).length();
916  reference += m_sectionConfigurations.at(sectionOfTheNewCursorPosition).suffix().length();
917  if (newPos <= reference) {
918  break;
919  }
920  }
921 
922  updatePrefixValueSuffixText();
923  setCurrentIndexWithoutUpdatingText(sectionOfTheNewCursorPosition);
924  q_pointer->lineEdit()->setText(m_textBeforeCurrentValue + m_textOfCurrentValue + m_textAfterCurrentValue);
925  int correctedCursorPosition = newPos;
926  if (cursorPositionHasToBeAdaptedToTextLenghtChange) {
927  correctedCursorPosition = newPos + q_pointer->lineEdit()->text().length() - oldTextLength;
928  }
929  q_pointer->lineEdit()->setCursorPosition(correctedCursorPosition);
930 
931  // Make sure that the buttons for step up and step down are updated.
932 
933  q_pointer->update();
934 }
