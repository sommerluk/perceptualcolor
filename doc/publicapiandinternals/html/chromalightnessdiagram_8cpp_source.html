136 {

177 LchDouble ChromaLightnessDiagram::ChromaLightnessDiagramPrivate::fromWidgetPixelPositionToColor(const QPoint widgetPixelPosition) const
178 {
179  const QPointF offset(leftBorderPhysical(), defaultBorderPhysical());
180  const QPointF imageCoordinatePoint = widgetPixelPosition
181  // Offset to pass from widget reference system to image reference system:
182  - offset / q_pointer->devicePixelRatioF()
183  // Offset to pass from pixel positions to coordinate points:
184  + QPointF(0.5, 0.5);
185  LchDouble color;
186  color.h = m_currentColor.h;
187  const qreal diagramHeight = calculateImageSizePhysical().height() / q_pointer->devicePixelRatioF();
188  if (diagramHeight > 0) {
189  color.l = imageCoordinatePoint.y() * 100.0 / diagramHeight * (-1.0) + 100.0;
190  color.c = imageCoordinatePoint.x() * 100.0 / diagramHeight;
191  } else {
192  color.l = 50;
193  color.c = 0;
194  }
195  return color;
196 }

267 void ChromaLightnessDiagram::paintEvent(QPaintEvent *event)
268 {
269  Q_UNUSED(event);
270  // We do not paint directly on the widget, but on a QImage buffer first:
271  // Render anti-aliased looks better. But as Qt documentation says:
272  //
273  // “Renderhints are used to specify flags to QPainter that may or
274  // may not be respected by any given engine.”
275  //
276  // Painting here directly on the widget might lead to different
277  // anti-aliasing results depending on the underlying window system. This
278  // is especially problematic as anti-aliasing might shift or not a pixel
279  // to the left or to the right. So we paint on a QImage first. As QImage
280  // (at difference to QPixmap and a QWidget) is independent of native
281  // platform rendering, it guarantees identical anti-aliasing results on
282  // all platforms. Here the quote from QPainter class documentation:
283  //
284  // “To get the optimal rendering result using QPainter, you should
285  // use the platform independent QImage as paint device; i.e. using
286  // QImage will ensure that the result has an identical pixel
287  // representation on any platform.”
288  QImage paintBuffer(physicalPixelSize(), QImage::Format_ARGB32_Premultiplied);
289  paintBuffer.fill(Qt::transparent);
290  QPainter painter(&paintBuffer);
291  QPen pen;
292 
293  // Paint the diagram itself as available in the cache.
294  painter.setRenderHint(QPainter::Antialiasing, false);
295  painter.drawImage(
296  // Operating in physical pixels:
297  d_pointer->leftBorderPhysical(), // x position (top-left)
298  d_pointer->defaultBorderPhysical(), // y position (top-left)
299  d_pointer->m_chromaLightnessImage.getImage() // image
300  );
301 
302  // Paint a focus indicator.
303  //
304  // We could paint a focus indicator (round or rectangular) around the
305  // handle. Depending on the currently selected hue for the diagram,
306  // it looks ugly because the colors of focus indicator and diagram
307  // do not harmonize, or it is mostly invisible the the colors are
308  // similar. So this approach does not work well.
309  //
310  // It seems better to paint a focus indicator for the whole widget.
311  // We could use the style primitives to paint a rectangular focus
312  // indicator around the whole widget:
313  //
314  // style()->drawPrimitive(
315  // QStyle::PE_FrameFocusRect,
316  // &option,
317  // &painter,
318  // this
319  // );
320  //
321  // However, this does not work well because the chroma-lightness
322  // diagram has usually a triangular shape. The style primitive, however,
323  // often paints just a line at the bottom of the widget. That does not
324  // look good. An alternative approach is that we paint ourselves a focus
325  // indicator only on the left of the diagram (which is the place of
326  // black/gray/white, so the won't be any problems with non-harmonic
327  // colors).
328  //
329  // Then we have to design the line that we want to display. It is better
330  // to do that ourselves instead of relying on generic QStyle::PE_Frame or
331  // similar solutions as their result seems to be quite unpredictable across
332  // various styles. So we use handleOutlineThickness as line width and
333  // paint it at the left-most possible position.
334  if (hasFocus()) {
335  pen = QPen();
336  pen.setWidthF(handleOutlineThickness() * devicePixelRatioF());
337  pen.setColor(focusIndicatorColor());
338  pen.setCapStyle(Qt::PenCapStyle::FlatCap);
339  painter.setPen(pen);
340  painter.setRenderHint(QPainter::Antialiasing, true);
341  const QPointF pointOne(
342  // x:
343  handleOutlineThickness() * devicePixelRatioF() / 2.0,
344  // y:
345  0 + d_pointer->defaultBorderPhysical());
346  const QPointF pointTwo(
347  // x:
348  handleOutlineThickness() * devicePixelRatioF() / 2.0,
349  // y:
350  physicalPixelSize().height() - d_pointer->defaultBorderPhysical());
351  painter.drawLine(pointOne, pointTwo);
352  }
353 
354  // Paint the handle on-the-fly.
355  const int diagramHeight = d_pointer->calculateImageSizePhysical().height();
356  QPointF colorCoordinatePoint = QPointF(
357  // x:
358  d_pointer->m_currentColor.c * diagramHeight / 100.0,
359  // y:
360  d_pointer->m_currentColor.l * diagramHeight / 100.0 * (-1) + diagramHeight);
361  colorCoordinatePoint += QPointF(d_pointer->leftBorderPhysical(), // horizontal offset
362  d_pointer->defaultBorderPhysical() // vertical offset
363  );
364  pen = QPen();
365  pen.setWidthF(handleOutlineThickness() * devicePixelRatioF());
366  pen.setColor(handleColorFromBackgroundLightness(d_pointer->m_currentColor.l));
367  painter.setPen(pen);
368  painter.setRenderHint(QPainter::Antialiasing, true);
369  painter.drawEllipse(colorCoordinatePoint, // center
370  handleRadius() * devicePixelRatioF(), // x radius
371  handleRadius() * devicePixelRatioF() // y radius
372  );
373 
374  // Paint the buffer to the actual widget
375  paintBuffer.setDevicePixelRatio(devicePixelRatioF());
376  QPainter widgetPainter(this);
377  widgetPainter.setRenderHint(QPainter::Antialiasing, true);
378  widgetPainter.drawImage(0, 0, paintBuffer);
379 }

402 void ChromaLightnessDiagram::keyPressEvent(QKeyEvent *event)
403 {
404  LchDouble temp = d_pointer->m_currentColor;
405  switch (event->key()) {
406  case Qt::Key_Up:
407  temp.l += singleStepLightness;
408  break;
409  case Qt::Key_Down:
410  temp.l -= singleStepLightness;
411  break;
412  case Qt::Key_Left:
413  temp.c -= singleStepChroma;
414  break;
415  case Qt::Key_Right:
416  temp.c += singleStepChroma;
417  break;
418  case Qt::Key_PageUp:
419  temp.l += pageStepLightness;
420  break;
421  case Qt::Key_PageDown:
422  temp.l -= pageStepLightness;
423  break;
424  case Qt::Key_Home:
425  temp.c += pageStepChroma;
426  break;
427  case Qt::Key_End:
428  temp.c -= pageStepChroma;
429  break;
430  default:
431  // Quote from Qt documentation:
432  //
433  // “If you reimplement this handler, it is very important that
434  // you call the base class implementation if you do not act
435  // upon the key.
436  //
437  // The default implementation closes popup widgets if the
438  // user presses the key sequence for QKeySequence::Cancel
439  // (typically the Escape key). Otherwise the event is
440  // ignored, so that the widget’s parent can interpret it.“
441  QWidget::keyPressEvent(event);
442  return;
443  }
444  // Here we reach only if the key has been recognized. If not, in the
445  // default branch of the switch statement, we would have passed the
446  // keyPressEvent yet to the parent and returned.
447 
448  // Set the new image coordinates (only takes effect when image
449  // coordinates are indeed different)
450  setCurrentColor(
451  // Search for the nearest color without changing the hue:
452  d_pointer->m_rgbColorSpace->nearestInGamutColorByAdjustingChromaLightness(temp));
453 }

470 bool ChromaLightnessDiagram::ChromaLightnessDiagramPrivate::isWidgetPixelPositionInGamut(const QPoint widgetPixelPosition) const
471 {
472  if (calculateImageSizePhysical().isEmpty()) {
473  // If there is no displayed gamut, the anser must be false.
474  // But fromWidgetPixelPositionToColor() would return an in-gamut
475  // fallback color nevertheless. Therefore, we have to catch
476  // the special case with an empty diagram here manually.
477  return false;
478  }
479 
480  const LchDouble color = fromWidgetPixelPositionToColor(widgetPixelPosition);
481  return (
482  // Test if C is in range. This is important because a negative C value
483  // can be in-gamut, but is not in the _displayed_ gamut.
484  isInRange<qreal>(0, color.c, m_rgbColorSpace->maximumChroma())
485  // Test for out-of-range lightness (mainly for performance reasons)
486  && isInRange<qreal>(0, color.l, 100)
487  // Test actually for in-gamut color
488  && m_rgbColorSpace->isInGamut(color));
489 }

499 void ChromaLightnessDiagram::setCurrentColor(const PerceptualColor::LchDouble &newCurrentColor)
500 {
501  if (newCurrentColor.hasSameCoordinates(d_pointer->m_currentColor)) {
502  return;
503  }
504 
505  double oldHue = d_pointer->m_currentColor.h;
506  d_pointer->m_currentColor = newCurrentColor;
507  if (d_pointer->m_currentColor.h != oldHue) {
508  // Update the diagram (only if the hue has changed):
509  d_pointer->m_chromaLightnessImage.setHue(d_pointer->m_currentColor.h);
510  }
511  update(); // Schedule a paint event
512  Q_EMIT currentColorChanged(newCurrentColor);
513 }

551 QSize ChromaLightnessDiagram::minimumSizeHint() const
552 {
553  const int minimumHeight = qRound(
554  // Top border and bottom border:
555  2.0 * d_pointer->defaultBorderPhysical() / devicePixelRatioF()
556  // Add the height for the diagram:
557  + gradientMinimumLength());
558  const int minimumWidth = qRound(
559  // Left border and right border:
560  (d_pointer->leftBorderPhysical() + d_pointer->defaultBorderPhysical()) / devicePixelRatioF()
561  // Add the gradient minimum length from y axis, multiplied with
562  // the factor to allow at correct scaling showing up the whole
563  // chroma range of the gamut.
564  + gradientMinimumLength() * d_pointer->m_rgbColorSpace->maximumChroma() / 100.0);
565  // Expand to the global minimum size for GUI elements
566  return QSize(minimumWidth, minimumHeight).expandedTo(QApplication::globalStrut());
567 }
