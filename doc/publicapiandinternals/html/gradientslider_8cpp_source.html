325 void GradientSlider::wheelEvent(QWheelEvent *event)
326 {
327  qreal steps = standardWheelStepCount(event);
328  // Only react on good old vertical wheels, and not on horizontal wheels
329  if (steps != 0) {
330  qreal stepSize;
331  if (QGuiApplication::keyboardModifiers().testFlag(Qt::ControlModifier) || QGuiApplication::keyboardModifiers().testFlag(Qt::ShiftModifier)) {
332  stepSize = pageStep();
333  } else {
334  stepSize = singleStep();
335  };
336  setValue(d_pointer->m_value + steps * stepSize);
337  } else {
338  // Donâ€™t accept the event and let it up to the default treatment:
339  event->ignore();
340  }
341 }

360 void GradientSlider::keyPressEvent(QKeyEvent *event)
361 {
362  switch (event->key()) {
363  case Qt::Key_Up:
364  case Qt::Key_Plus:
365  setValue(d_pointer->m_value + d_pointer->m_singleStep);
366  break;
367  case Qt::Key_Down:
368  case Qt::Key_Minus:
369  setValue(d_pointer->m_value - d_pointer->m_singleStep);
370  break;
371  case Qt::Key_Left:
372  if (layoutDirection() == Qt::LayoutDirection::LeftToRight) {
373  setValue(d_pointer->m_value - d_pointer->m_singleStep);
374  } else {
375  setValue(d_pointer->m_value + d_pointer->m_singleStep);
376  }
377  break;
378  case Qt::Key_Right:
379  if (layoutDirection() == Qt::LayoutDirection::LeftToRight) {
380  setValue(d_pointer->m_value + d_pointer->m_singleStep);
381  } else {
382  setValue(d_pointer->m_value - d_pointer->m_singleStep);
383  }
384  break;
385  case Qt::Key_PageUp:
386  setValue(d_pointer->m_value + d_pointer->m_pageStep);
387  break;
388  case Qt::Key_PageDown:
389  setValue(d_pointer->m_value - d_pointer->m_pageStep);
390  break;
391  case Qt::Key_Home:
392  setValue(0);
393  break;
394  case Qt::Key_End:
395  setValue(1);
396  break;
397  default:
398  /* Quote from Qt documentation:
399  *
400  * â€œIf you reimplement this handler, it is very important that
401  * you call the base class implementation if you do not act
402  * upon the key.
403  *
404  * The default implementation closes popup widgets if the
405  * user presses the key sequence for QKeySequence::Cancel
406  * (typically the Escape key). Otherwise the event is
407  * ignored, so that the widgetâ€™s parent can interpret it.â€œ */
408  QWidget::keyPressEvent(event);
409  }
410 }

493 qreal GradientSlider::GradientSliderPrivate::fromWidgetPixelPositionToValue(QPoint pixelPosition)
494 {
495  // We are interested in the point in the middle of the given pixel.
496  const QPointF coordinatePoint = pixelPosition + QPointF(0.5, 0.5);
497  qreal temp;
498  if (m_orientation == Qt::Orientation::Vertical) {
499  temp = (q_pointer->size().height() - coordinatePoint.y()) / static_cast<qreal>(q_pointer->size().height());
500  } else {
501  if (q_pointer->layoutDirection() == Qt::LayoutDirection::LeftToRight) {
502  temp = coordinatePoint.x() / static_cast<qreal>(q_pointer->size().width());
503  } else {
504  temp = (q_pointer->size().width() - coordinatePoint.x()) / static_cast<qreal>(q_pointer->size().width());
505  }
506  }
507  return qBound<qreal>(0, temp, 1);
508 }

515 void GradientSlider::paintEvent(QPaintEvent *event)
516 {
517  Q_UNUSED(event);
518  // We do not paint directly on the widget, but on a QImage buffer first:
519  // Render anti-aliased looks better. But as Qt documentation says:
520  //
521  // â€œRenderhints are used to specify flags to QPainter that may or
522  // may not be respected by any given engine.â€
523  //
524  // Painting here directly on the widget might lead to different
525  // anti-aliasing results depending on the underlying window system. This
526  // is especially problematic as anti-aliasing might shift or not a pixel
527  // to the left or to the right. So we paint on a QImage first. As QImage
528  // (at difference to QPixmap and a QWidget) is independent of native
529  // platform rendering, it guarantees identical anti-aliasing results on
530  // all platforms. Here the quote from QPainter class documentation:
531  //
532  // â€œTo get the optimal rendering result using QPainter, you should
533  // use the platform independent QImage as paint device; i.e. using
534  // QImage will ensure that the result has an identical pixel
535  // representation on any platform.â€
536  QImage paintBuffer;
537 
538  // Paint the gradient itself.
539  // Make sure the image will be correct. We set length and thicknes,
540  // just to be sure (we might have missed a resize event). Also,
541  // the device pixel ratio float might have changed because the
542  // window has been moved to another screen. We do not update the
543  // first and the second color because we have complete control
544  // about these values and are sure the any changes have yet been
545  // applied.
546  d_pointer->m_gradientImageCache.setDevicePixelRatioF(devicePixelRatioF());
547  d_pointer->m_gradientImageCache.setGradientLength(d_pointer->physicalPixelLength());
548  d_pointer->m_gradientImageCache.setGradientThickness(
549  // Normally, this should not change, but maybe on Hight-DPI
550  // devices there are some differences.
551  d_pointer->physicalPixelLength());
552  paintBuffer = d_pointer->m_gradientImageCache.getImage();
553 
554  // Draw slider handle
555  QPainter bufferPainter(&paintBuffer);
556  // We use antialiasing. As our current handle is just a horizontal or
557  // vertical line, it might be slightly sharper without antialiasing.
558  // But all other widgets of this library WILL USE antialiasing because
559  // their handles are not perfectly horizontal or vertical and without
560  // antialiasing they might look terrible. Now, when antialiasing is NOT
561  // used, the line thickness is rounded. This would lead to a different
562  // thickness in this widget compared to the other widgets. This is not
563  // a good idea. Therefore, we USE antialiasing here. Anyway, in practical
564  // tests, it seems almost as sharp as without antialiasing, and
565  // additionally the position is more exact!
566  bufferPainter.setRenderHint(QPainter::Antialiasing, true);
567  QPen pen;
568  const qreal handleCoordinatePoint = d_pointer->physicalPixelLength() / devicePixelRatioF() * d_pointer->m_value;
569  if (hasFocus()) {
570  pen.setWidthF(handleOutlineThickness() * 3);
571  pen.setColor(focusIndicatorColor());
572  bufferPainter.setPen(pen);
573  bufferPainter.drawLine(QPointF(handleCoordinatePoint, 0), QPointF(handleCoordinatePoint, gradientThickness()));
574  }
575  pen.setWidthF(handleOutlineThickness());
576  pen.setColor(handleColorFromBackgroundLightness(d_pointer->m_gradientImageCache.colorFromValue(d_pointer->m_value).l));
577  bufferPainter.setPen(pen);
578  bufferPainter.drawLine(QPointF(handleCoordinatePoint, 0), QPointF(handleCoordinatePoint, gradientThickness()));
579 
580  // Paint the buffer to the actual widget
581  QTransform transform;
582  // The m_gradientImageCache contains the gradient always
583  // in a default form, independant of the actual orientation
584  // of this widget and independant of its actual layout direction:
585  // In the default form, the first color is always on the left, and the
586  // second color is always on the right. To paint it, we have to
587  // rotate it if our actual orientation is vertical. And we have to
588  // mirror it when our actual layout direction is RTL.
589  if (d_pointer->m_orientation == Qt::Orientation::Vertical) {
590  if (layoutDirection() == Qt::LayoutDirection::RightToLeft) {
591  // Even on vertical gradients, we mirror the image, so that
592  // the well-aligned edge of the transparency background is
593  // always aligned according to the writing direction.
594  transform.scale(-1, 1);
595  transform.rotate(270);
596  transform.translate(size().height() * (-1), size().width() * (-1));
597  } else {
598  transform.rotate(270);
599  transform.translate(size().height() * (-1), 0);
600  }
601  } else {
602  if (layoutDirection() == Qt::LayoutDirection::RightToLeft) {
603  transform.scale(-1, 1);
604  transform.translate(size().width() * (-1), 0);
605  }
606  }
607  QPainter widgetPainter(this);
608  widgetPainter.setTransform(transform);
609  widgetPainter.drawImage(0, 0, paintBuffer);
610 
611  // // TODO Draw a focus rectangle like this?:
612  // widgetPainter.setTransform(QTransform());
613  // if (hasFocus()) {
614  // QStyleOptionFocusRect opt;
615  // opt.palette = palette();
616  // opt.rect = rect();
617  // opt.state = QStyle::State_None |
618  // QStyle::State_KeyboardFocusChange; style()->drawPrimitive(
619  // QStyle::PE_FrameFocusRect,
620  // &opt,
621  // &widgetPainter,
622  // this
623  // );
624  // }
625 }
