40 RgbColorSpace::RgbColorSpace(QObject *parent) : QObject(parent)
41 {
42  // Create an ICC v4 profile object for the Lab color space.
43  // NULL means: Default white point (D50)
44  // TODO Does this make sense? sRGB white point is D65!
45  cmsHPROFILE labProfileHandle = cmsCreateLab4Profile(NULL);
46  // Create an ICC profile object for the sRGB color space.
47  cmsHPROFILE rgbProfileHandle = cmsCreate_sRGBProfile();
48  m_description = getInformationFromProfile(
49  rgbProfileHandle,
50  cmsInfoDescription
51  );
52  // TODO Only change the description to "sRGB" if the build-in sRGB is
53  // used, not when an actual external ICC profile is used.
54  m_description = tr("sRGB");
55  // Create the transforms
56  m_transformLabToRgbHandle = cmsCreateTransform(
57  labProfileHandle, // input profile handle
58  TYPE_Lab_DBL, // input buffer format
59  rgbProfileHandle, // output profile handle
60  TYPE_RGB_DBL, // output buffer format
61  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
62  0 // flags
63  );
64  m_transformLabToRgb16Handle = cmsCreateTransform(
65  labProfileHandle, // input profile handle
66  TYPE_Lab_DBL, // input buffer format
67  rgbProfileHandle, // output profile handle
68  TYPE_RGB_16, // output buffer format
69  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
70  0 // flags
71  );
72  m_transformRgbToLabHandle = cmsCreateTransform(
73  rgbProfileHandle, // input profile handle
74  TYPE_RGB_DBL, // input buffer format
75  labProfileHandle, // output profile handle
76  TYPE_Lab_DBL, // output buffer format
77  INTENT_ABSOLUTE_COLORIMETRIC, // rendering intent
78  0 // flags
79  );
80  // Close profile (free memory)
81  cmsCloseProfile(labProfileHandle);
82  cmsCloseProfile(rgbProfileHandle);
83 
84  // Now we know for sure that lowerChroma is in-gamut and upperChroma is
85  // out-of-gamutâ€¦
86  cmsCIELCh candidate;
87  candidate.L = 0;
88  candidate.C = 0;
89  candidate.h = 0;
90  while (!inGamut(candidate) && (candidate.L < 100)) {
91  candidate.L += Helper::gamutPrecision;
92  }
93  m_blackpointL = candidate.L;
94  candidate.L = 100;
95  while (!inGamut(candidate) && (candidate.L > 0)) {
96  candidate.L -= Helper::gamutPrecision;
97  }
98  m_whitepointL = candidate.L;
99  if (m_whitepointL <= m_blackpointL) {
100  qCritical() << "Unable to find blackpoint and whitepoint on gray axis.";
101  throw 0;
102  }
103 }

163 QColor RgbColorSpace::colorRgb(const cmsCIELab &Lab) const
164 {
165  QColor temp; // By default, without initialization this is an invalid color
166  Helper::cmsRGB rgb;
167  cmsDoTransform(m_transformLabToRgbHandle, &Lab, &rgb, 1); // convert exactly 1 value
168  if (Helper::inRange<cmsFloat64Number>(0, rgb.red, 1) &&
169  Helper::inRange<cmsFloat64Number>(0, rgb.green, 1) &&
170  Helper::inRange<cmsFloat64Number>(0, rgb.blue, 1)) {
171  // We are within the gamut
172  temp = QColor::fromRgbF(rgb.red, rgb.green, rgb.blue);
173  }
174  return temp;
175 }

191 Helper::cmsRGB RgbColorSpace::colorRgbBoundSimple(const cmsCIELab &Lab) const
192 {
193  cmsUInt16Number rgb_int[3];
194  cmsDoTransform(m_transformLabToRgb16Handle, &Lab, rgb_int, 1); // convert exactly 1 value
195  Helper::cmsRGB temp;
196  temp.red = rgb_int[0] / static_cast<qreal>(65535);
197  temp.green = rgb_int[1] / static_cast<qreal>(65535);
198  temp.blue = rgb_int[2] / static_cast<qreal>(65535);
199  return temp;
200 }

289 QString RgbColorSpace::getInformationFromProfile(
290  cmsHPROFILE profileHandle,
291  cmsInfoType infoType
292 )
293 {
294  // Initialize a char array of 3 values (two actual characters and a
295  // terminating null)
296  // The recommended default value for language following LittleCMS
297  // documentation is "en".
298  char languageCode[3] = "en";
299  // The recommended default value for country following LittleCMS
300  // documentation is "US".
301  char countryCode[3] = "US";
302  // Update languageCode and countryCode to the actual locale (if possible)
303  QStringList list = QLocale().name().split('_');
304  // The locale codes should be ASCII only, so QString::toLatin1() should
305  // return valid results.
306  if (list.at(0).size() == 2) {
307  languageCode[0] = list.at(0).at(0).toLatin1();
308  languageCode[1] = list.at(0).at(1).toLatin1();
309  // No need for languageCode[2] = 0; for null-terminated string,
310  // because the array was yet initialized
311  if (list.at(1).size() == 2) {
312  countryCode[0] = list.at(1).at(0).toLatin1();
313  countryCode[1] = list.at(1).at(1).toLatin1();
314  // No need for countryCode[2] = 0; for null-terminated string,
315  // because the array was yet initialized
316  }
317  }
318 
319  // Calculate the size of the buffer that we have to provide for
320  // cmsGetProfileInfo in order to return a value.
321  cmsUInt32Number resultLength = cmsGetProfileInfo(
322  // profile in which we search
323  profileHandle,
324  // the type of information we search
325  infoType,
326  // the preferred language in which we want to get the information
327  languageCode,
328  // the preferred country for which we want to get the information
329  countryCode,
330  // do not actually return the information,
331  // just calculate the required buffer size
332  NULL,
333  // do not actually return the information,
334  // just calculate the required buffer size
335  0
336  );
337  // For the actual buffer size, increment by 1. This helps us to
338  // guarantee a null-terminated string later on.
339  cmsUInt32Number bufferLength = resultLength + 1;
340 
341  // Allocate the buffer
342  wchar_t *buffer = new wchar_t [bufferLength];
343  // Initialize the buffer with NULL
344  for (int i = 0; i < bufferLength - 1; ++i) {
345  *(buffer + i) = 0;
346  }
347 
348  // Write the actual information to the buffer
349  cmsGetProfileInfo(
350  // profile in which we search
351  profileHandle,
352  // the type of information we search
353  infoType,
354  // the preferred language in which we want to get the information
355  languageCode,
356  // the preferred country for which we want to get the information
357  countryCode,
358  // the buffer into which the requested information will be written
359  buffer,
360  // the buffer size as previously calculated
361  resultLength
362  );
363  // Make absolutely sure the buffer is null-terminated by marking its last
364  // element (the one that was the +1 "extra" element) as NULL.
365  *(buffer + (bufferLength - 1)) = 0;
366 
367  // Create a QString() from the from the buffer
368  /* cmsGetProfileInfo returns often strings that are smaller than the
369  * previously calculated buffer size. But it seems they are
370  * null-terminated strings. */
371  QString result = QString::fromWCharArray(
372  buffer, // read from the buffer
373  -1 // read until the first NULL element
374  );
375 
376  // Free allocated memory of the buffer
377  delete [] buffer;
378 
379  // Return
380  return result;
381 }
