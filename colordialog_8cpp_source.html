110 void ColorDialog::setCurrentColor(const QColor& color)
111 {
112  QColor temp;
113  if (color.isValid()) {
114  temp = color;
115  // QColorDialog would instead call QColor.rgb() which
116  // rounds to 8 bit per channel.
117  } else {
118  // For invalid colors same behaviour as QColorDialog
119  temp = QColor(Qt::black);
120  }
121  if (testOption(QColorDialog::ColorDialogOption::ShowAlphaChannel)) {
122  m_alphaSelector->setAlpha(color.alphaF());
123  } else {
124  m_alphaSelector->setAlpha(1);
125  }
126  setCurrentOpaqueColor(FullColorDescription(m_rgbColorSpace, temp));
127 }

176 void ColorDialog::setCurrentOpaqueColor(const FullColorDescription& color)
177 {
178  if (m_isColorChangeInProgress
179  || (!color.isValid())
180  || (color == m_currentOpaqueColor)
181  ) {
182  // Nothing to do!
183  return;
184  }
185 
186  // If we have really work to do, block recursive calls of this function
187  m_isColorChangeInProgress = true;
188 
189  // Save currentColor() for later comparision
190  // Using currentColor() makes sure correct alpha treatment!
191  QColor oldQColor = currentColor();
192 
193  // Update m_currentOpaqueColor
194  m_currentOpaqueColor = color;
195 
196  // Update all the widgets for opaque color…
197  QColor tempRgbQColor = color.toRgbQColor();
198  tempRgbQColor.setAlpha(255);
199  m_rgbRedSpinbox->setValue(tempRgbQColor.redF() * 255);
200  m_rgbGreenSpinbox->setValue(tempRgbQColor.greenF() * 255);
201  m_rgbBlueSpinbox->setValue(tempRgbQColor.blueF() * 255);
202  m_hsvHueSpinbox->setValue(color.toHsvQColor().hsvHueF() * 360);
203  m_hsvSaturationSpinbox->setValue(
204  color.toHsvQColor().hsvSaturationF() * 255
205  );
206  m_hsvValueSpinbox->setValue(color.toHsvQColor().valueF() * 255);
207  m_hlcLineEdit->setText(textForHlcLineEdit());
208  m_rgbLineEdit->setText(m_currentOpaqueColor.toRgbHexString());
209  m_lchLightnessSelector->setFraction(
210  color.toLch().L / static_cast<qreal>(100)
211  );
212  m_chromaHueDiagram->setColor(color);
213  m_wheelColorPicker->setCurrentColor(m_currentOpaqueColor);
214  m_alphaSelector->setColor(m_currentOpaqueColor);
215 
216  // Update widgets that take alpha information
217  updateColorPatch();
218 
219  // Emit signal currentColorChanged() only if necessary
220  if (currentColor() != oldQColor) {
221  Q_EMIT currentColorChanged(currentColor());
222  }
223 
224  // End of this function. Unblock resursive
225  // function calls before returning.
226  m_isColorChangeInProgress = false;
227 }

488 void ColorDialog::handleFocusChange(QWidget *old, QWidget *now)
489 {
490  if (old == m_hlcLineEdit) {
491  if (m_hlcLineEdit->text() != textForHlcLineEdit()) {
492  // Only read-in the values if really changed. That's important
493  // because otherwise, passing just with the focus through the
494  // m_hlcLineEdit widget, each time the value would be read-in.
495  // This would lead to rounding errors: L might be 50.8 before,
496  // but displayed is just 51. At read-in, it would be changed from
497  // 50.8 to 51, which is undesirable except the user has really
498  // changed a value in this widget.
499  readHlcNumericValues();
500  }
501  }
502 }

506 void ColorDialog::readHlcNumericValues()
507 {
508  cmsCIELCh lch;
509  QStringList temp = m_hlcLineEdit->text().split(QStringLiteral(u" "));
510  if (temp.size() == 3) {
511  lch.h = temp.at(0).toInt();
512  lch.L = qMin(temp.at(1).toInt(), 100);
513  lch.C = temp.at(2).toInt();
514  setCurrentOpaqueColor(
515  FullColorDescription(
516  m_rgbColorSpace,
517  lch,
518  FullColorDescription::outOfGamutBehaviour::sacrifyChroma
519  )
520  );
521  } else {
522  m_hlcLineEdit->setText(textForHlcLineEdit());
523  }
524 }

811 void ColorDialog::applyLayoutDimensions()
812 {
813  bool collapsedLayout; // true for small layout, false for large layout.
814  int effectivelyAvailableScreenWidth;
815  int widthThreeshold;
816  switch (m_layoutDimensions) {
817  case DialogLayoutDimensions::collapsed:
818  collapsedLayout = true;
819  break;
820  case DialogLayoutDimensions::expanded:
821  collapsedLayout = false;
822  break;
823  case DialogLayoutDimensions::automatic:
