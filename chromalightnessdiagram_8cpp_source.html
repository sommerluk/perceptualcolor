232 void ChromaLightnessDiagram::paintEvent(QPaintEvent* event)
233 {
234  Q_UNUSED(event);
235  // We do not paint directly on the widget, but on a QImage buffer first:
236  // Render anti-aliased looks better. But as Qt documentation says:
237  //
238  // “Renderhints are used to specify flags to QPainter that may or
239  // may not be respected by any given engine.”
240  //
241  // Painting here directly on the widget might lead to different
242  // anti-aliasing results depending on the underlying window system. This is
243  // especially problematic as anti-aliasing might shift or not a pixel to the
244  // left or to the right. So we paint on a QImage first. As QImage (at
245  // difference to QPixmap and a QWidget) is independant of native platform
246  // rendering, it guarantees identical anti-aliasing results on all
247  // platforms. Here the quote from QPainter class documentation:
248  //
249  // “To get the optimal rendering result using QPainter, you should use
250  // the platform independent QImage as paint device; i.e. using QImage
251  // will ensure that the result has an identical pixel representation
252  // on any platform.”
253  QImage paintBuffer(size(), QImage::Format_ARGB32);
254  paintBuffer.fill(Qt::transparent);
255  QPainter painter(&paintBuffer);
256 
257  QPen pen;
258 
259  // Paint the diagram itself as available in the cache.
260  updateDiagramCache();
261  painter.drawImage(m_border, m_border, m_diagramImage);
262 
263  /* Paint a focus indicator.
264  *
265  * We could paint a focus indicator (round or rectangular) around the marker.
266  * Depending on the currently selected hue for the diagram, it looks ugly
267  * because the colors of focus indicator and diagram do not harmonize, or
268  * it is mostly invisible the the colors are similar. So this apporach does
269  * not work well.
270  *
271  * It seems better to paint a focus indicator for the whole widget.
272  * We could use the style primitives to paint a rectangular focus indicator
273  * around the whole widget:
274  * style()->drawPrimitive(QStyle::PE_FrameFocusRect, &option, &painter, this);
275  * However, this does not work well because the chroma-lightness diagram has
276  * usually a triangular shape. The style primitive, however, often paints
277  * just a line at the bottom of the widget. That does not look good. An
278  * alternative approach is that we paint ourself a focus indicator only
279  * on the left of the diagram (which is the place of black/grey/white,
280  * so the won't be any problems with non-harmonic colors). We can
281  * get the color from the current style:
282  * QColor hightlightColor = palette().color(QPalette::Highlight);
283  * QBrush highlightBrush = palette().highlight();
284  * Then we have to design the line that we want to display. It is better to
285  * do that ourself instead of relying on generic QStyle::PE_Frame or similar
286  * solutions as their result seems to be quite unpredictible accross various
287  * styles. So we use markerThickness as line width and paint it at the
288  * left-most possible position. As the border() property accomodates also to
289  * markerRadius, the distance of the focus line to the real diagram also
290  * does, which looks nice.
291  */
292  if (hasFocus()) {
293  pen.setWidth(markerThickness);
294  pen.setColor(
295  palette().color(QPalette::Highlight)
296  );
297  painter.setPen(pen);
298  painter.drawLine(
299  markerThickness / 2, // 0.5 is rounded down to 0.0
300  0 + m_border,
301  markerThickness / 2, // 0.5 is rounded down to 0.0
302  size().height() - m_border
303  );
304  }
305 
306  // Paint the marker on-the-fly.
307  // Render anti-aliased looks better. But as Qt documentation says: “Renderhints are used to
308  // specify flags to QPainter that may or may not be respected by any given engine.” Now, we
309  // are painting here directly on the widget, which might lead to different anti-aliasing
310  // results depending on the underlying window system. An alternative approach might be to
311  // do the rendering on a QImage first. As QImage (at difference to QPixmap and widgets) is
312  // independant of native platform rendering, it would garantee identical results on all
313  // platforms. But it seems a litte overkill, so we don't do that here. Anyway here the
314  // quote from QPainter class documentation:
315  //
316  // To get the optimal rendering result using QPainter, you should use the platform independent
317  // QImage as paint device; i.e. using QImage will ensure that the result has an identical pixel
318  // representation on any platform.
319  painter.setRenderHint(QPainter::Antialiasing);
320  QPoint imageCoordinates = currentImageCoordinates();
321  pen.setWidth(markerThickness);
322  int markerBackgroundLightness = m_color.toLch().L; // range: 0..100
323  if (markerBackgroundLightness >= 50) {
324  pen.setColor(Qt::black);
325  } else {
326  pen.setColor(Qt::white);
327  }
328  painter.setPen(pen);
329  painter.drawEllipse(
330  imageCoordinates.x() + m_border - markerRadius,
331  imageCoordinates.y() + m_border - markerRadius,
332  2 * markerRadius + 1,
333  2 * markerRadius + 1
334  );
335 
336  // Paint the buffer to the actual widget
337  QPainter(this).drawImage(0, 0, paintBuffer);
338 }

375 void ChromaLightnessDiagram::keyPressEvent(QKeyEvent *event)
376 {
377  // TODO singleStep & pageStep for ALL graphical widgets expressed in LCh
378  // values, not in pixel. Use values as inherited from base class!
379 
380  QPoint newImageCoordinates = currentImageCoordinates();
381  updateDiagramCache();
382  switch (event->key()) {
383  case Qt::Key_Up:
384  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(0, -1))) {
385  newImageCoordinates += QPoint(0, -1);
386  }
387  break;
388  case Qt::Key_Down:
389  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(0, 1))) {
390  newImageCoordinates += QPoint(0, 1);
391  }
392  break;
393  case Qt::Key_Left:
394  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(-1, 0))) {
395  newImageCoordinates += QPoint(-1, 0);
396  }
397  break;
398  case Qt::Key_Right:
399  if (imageCoordinatesInGamut(newImageCoordinates + QPoint(1, 0))) {
400  newImageCoordinates += QPoint(1, 0);
401  }
402  break;
403  case Qt::Key_PageUp:
404  newImageCoordinates.setY(0);
405  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(0, 1))) {
406  newImageCoordinates += QPoint(0, 1);
407  }
408  break;
409  case Qt::Key_PageDown:
410  newImageCoordinates.setY(m_diagramImage.height() - 1);
411  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(0, -1))) {
412  newImageCoordinates += QPoint(0, -1);
413  }
414  break;
415  case Qt::Key_Home:
416  newImageCoordinates.setX(0);
417  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(1, 0))) {
418  newImageCoordinates += QPoint(1, 0);
419  }
420  break;
421  case Qt::Key_End:
422  newImageCoordinates.setX(m_diagramImage.width() - 1);
423  while (!imageCoordinatesInGamut(newImageCoordinates + QPoint(-1, 0))) {
424  newImageCoordinates += QPoint(-1, 0);
425  }
426  break;
427  default:
428  /* Quote from Qt documentation:
429  *
430  * If you reimplement this handler, it is very important that you call the base class
431  * implementation if you do not act upon the key.
432  *
433  * The default implementation closes popup widgets if the user presses the key sequence
434  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
435  * so that the widget's parent can interpret it.
436  */
437  QWidget::keyPressEvent(event);
438  return;
439  }
440  // Here we reach only if the key has been recognized. If not, in the default branch of the
441  // switch statement, we would have passed the keyPressEvent yet to the parent and returned.
442  // Set the new image coordinates (only takes effect when image coordinates are indeed different)
443  setImageCoordinates(newImageCoordinates);
444 }

654 QImage ChromaLightnessDiagram::generateDiagramImage(
655  const qreal imageHue,
656  const QSize imageSize) const
657 {
658 QElapsedTimer myTimer;
659 myTimer.start();
660  cmsCIELCh LCh; // uses cmsFloat64Number internally
661  QColor rgbColor;
662  int x;
663  int y;
664  QImage temp_image = QImage(imageSize, QImage::Format_ARGB32);
665  const int maxHeight = imageSize.height() - 1;
666  const int maxWidth = imageSize.width() - 1;
667 
668  // Test if image size is too small.
669  if ((maxHeight < 1) || (maxWidth < 1)) {
670  // TODO How to react correctly here? Exception?
671  // maxHeight and maxWidth must be at least >= 1 for our algorithm. If they are 0, this would crash (division by 0).
672  return temp_image;
673  }
674 
675  // Initialize the image with transparency.
676  temp_image.fill(Qt::transparent);
677 
678  // Paint the gamut.
679  LCh.h = PolarPointF::normalizedAngleDegree(imageHue);
680  for (y = 0; y <= maxHeight; ++y) {
681  LCh.L = y * static_cast<cmsFloat64Number>(100) / maxHeight; // floating point division thanks to 100 which is a "cmsFloat64Number"
682  for (x = 0; x <= maxWidth; ++x) {
683  // Using the same scale as on the y axis. floating point
684  // division thanks to 100 which is a "cmsFloat64Number"
685  LCh.C = x * static_cast<cmsFloat64Number>(100) / maxHeight;
686  rgbColor = m_rgbColorSpace->colorRgb(LCh);
687  if (rgbColor.isValid()) {
688  // The pixel is within the gamut
689  temp_image.setPixelColor(
690  x,
691  maxHeight - y,
692  rgbColor
693  );
694  /* If color is out-of-gamut: We have chroma on the x axis and
695  * lightness on the y axis. We are drawing the pixmap line per
696  * line, so we go for given lightness from low chroma to high
697  * chroma. Because of the nature of most gamuts, if once in a
698  * line we have an out-of-gamut value, all other pixels that
699  * are more at the right will be out-of-gamut also. So we
700  * could optimize our code and break here. But as we are not
701  * sure about this (we do not know the gamut at compile time)
702  * for the moment we do not optimize the code.
703  */
704  }
705  }
706  }
707 
708 qDebug() << "Generating chroma-lightness gamut image took" << myTimer.restart() << "ms.";
709  return temp_image;
710 }

761 QPoint ChromaLightnessDiagram::nearestNeighborSearch(
762  const QPoint originalPoint,
763  const QImage &image
764 ) {
765  // test for special case: originalPoint itself is within the image and non-transparent
766  if (image.valid(originalPoint)) {
767  if (image.pixelColor(originalPoint).alpha() == 255) {
768  return originalPoint;
769  }
770  }
771 
772  // No special case. So we have to actually perfor a nearest-neighbor-search.
773  int x;
774  int y;
775  int currentBestX;
776  int currentBestY;
777  int currentBestDistanceSquare = std::numeric_limits<int>::max();
778  int temp;
779  for (x = 0; x < image.width(); x++) {
780  for (y = 0; y < image.height(); y++) {
781  if (image.pixelColor(x, y).alpha() == 255) {
782  temp = pow(originalPoint.x() - x, 2) + pow(originalPoint.y() - y, 2);
783  if (temp < currentBestDistanceSquare) {
784  currentBestX = x;
785  currentBestY = y;
786  currentBestDistanceSquare = temp;
787  }
788  }
789  }
790  }
791  if (currentBestDistanceSquare == std::numeric_limits<int>::max()) {
792  return QPoint(0, 0);
793  } else {
794  return QPoint(currentBestX, currentBestY);
795  }
796 }
