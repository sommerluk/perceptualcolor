87 void ColorPatch::paintEvent(QPaintEvent *event)
88 {
89  // Paint the frame, as provided by the base class
90  QFrame::paintEvent(event);
91 
92  // We do not paint directly on the widget, but on a QImage buffer first:
93  // As Qt documentation says:
94  //
95  // “To get the optimal rendering result using QPainter, you should
96  // use the platform independent QImage as paint device; i.e. using
97  // QImage will ensure that the result has an identical pixel
98  // representation on any platform.”
99  if (m_color.isValid()) {
100  QImage paintBuffer(size(), QImage::Format_ARGB32);
101  paintBuffer.fill(Qt::transparent);
102  QPainter painter(&paintBuffer);
103  if (m_color.alphaF() < 1) {
104  // Background for colors that are not fully opaque
105  painter.fillRect(
106  contentsRect(),
107  QBrush(Helper::transparencyBackground())
108  );
109  }
110  painter.fillRect(contentsRect(), m_color);
111  // Paint the buffer to the actual widget
112  QPainter(this).drawImage(0, 0, paintBuffer);
113  }
114 }
