80 FullColorDescription::FullColorDescription(RgbColorSpace *colorSpace, QColor color)
81 {
82  if (!color.isValid()) {
83  m_valid = false;
84  return;
85  }
86  switch (color.spec()) {
87  case QColor::Spec::Hsv:
88  m_hsvQColor = color;
89  m_rgbQColor = color.toRgb();
90  break;
91  case QColor::Spec::Rgb:
92  m_rgbQColor = color;
93  m_hsvQColor = color.toHsv();
94  break;
95  default:
96  m_rgbQColor = color.toRgb();
97  m_hsvQColor = color.toHsv();
98  break;
99  }
100  m_rgb.red = m_rgbQColor.redF();
101  m_rgb.green = m_rgbQColor.greenF();
102  m_rgb.blue = m_rgbQColor.blueF();
103  m_lab = colorSpace->colorLab(m_rgbQColor);
104  m_lch = toLch(m_lab);
105  m_alpha = color.alphaF();
106  m_valid = true;
107 }

118 FullColorDescription::FullColorDescription(RgbColorSpace *colorSpace, const cmsCIELab &lab, outOfGamutBehaviour behaviour, qreal alpha)
119 {
120  m_lch = toLch(lab);
121  normalizeLch();
122  if (behaviour == outOfGamutBehaviour::sacrifyChroma) {
123  moveChromaIntoGamut(colorSpace);
124  }
125  m_lab = toLab(m_lch);
126  m_rgb = colorSpace->colorRgbBoundSimple(m_lab);

128  m_hsvQColor = m_rgbQColor.toHsv();
129  m_alpha = alpha;
130  m_valid = true;

166 void FullColorDescription::moveChromaIntoGamut(RgbColorSpace *colorSpace)
167 {
168  // Test special case: If we are yet in-gamut…
169  if (colorSpace->inGamut(m_lch)) {
170  return;
171  }
172 
173  // Now we know: We are out-of-gamut…
174  cmsCIELCh lowerChroma {m_lch.L, 0, m_lch.h};
175  cmsCIELCh upperChroma {m_lch};
176  cmsCIELCh candidate;
177  if (colorSpace->inGamut(lowerChroma)) {
178  // Now we know for sure that lowerChroma is in-gamut and upperChroma is out-of-gamut…
179  candidate = upperChroma;
180  while (upperChroma.C - lowerChroma.C > Helper::gamutPrecision) {
181  // Our test candidate is half the way between lowerChroma and upperChroma:
182  candidate.C = (
183  (lowerChroma.C + upperChroma.C) / 2
184  );
185  if (colorSpace->inGamut(candidate)) {
186  lowerChroma = candidate;
187  } else {
188  upperChroma = candidate;
189  }
190  }
191  m_lch = lowerChroma;
192  } else {
193  if (m_lch.L < colorSpace->blackpointL()) {
194  m_lch.L = colorSpace->blackpointL();
195  m_lch.C = 0;
196  } else {
197  if (m_lch.L > colorSpace->whitepointL()) {
198  m_lch.L = colorSpace->blackpointL();
199  m_lch.C = 0;
200  }
201  }
202  }
203 }

330 QString FullColorDescription::toRgbHexString() const
331 {
332  // Format of the numbers:
333  // - Minimal field width: 2
334  // - The base of the number represenation: 16 (hexadecimal)
335  // - The fill character (leading zero): QLatin1Char('0')
336  return QStringLiteral(u"#%1%2%3")
337  .arg(qRound(m_rgb.red * 255), 2, 16, QLatin1Char('0'))
338  .arg(qRound(m_rgb.green * 255), 2, 16, QLatin1Char('0'))
339  .arg(qRound(m_rgb.blue * 255), 2, 16, QLatin1Char('0'))
340  .toUpper();
341 }
