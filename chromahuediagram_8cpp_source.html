137 void ChromaHueDiagram::mousePressEvent(QMouseEvent *event)
138 {
139  // TODO Also accept out-of-gamut clicks when they are covered by the
140  // current marker.
141  if (areImageCoordinatesWithinDiagramSurface(event->pos())) {
142  // Mouse focus is handeled manually because so we can accept
143  // focus only on mouse clicks within the displayed gamut, while
144  // rejecting focus otherwise. In the constructor, therefore
145  // Qt::FocusPolicy::TabFocus is specified, so that manual handling
146  // of mouse focus is up to this code here.
147  setFocus(Qt::MouseFocusReason);
148  // Enable mouse tracking from now on:
149  m_isMouseEventActive = true;
150  // As clicks are only accepted within the visible gamut, the mouse
151  // cursor is made invisible. Its function is taken over by the value
152  // marker within the displayed gamut.
153  setCursor(Qt::BlankCursor);
154  // Set the color property
155  setColorFromImageCoordinates(event->pos());
156  // Schedule a paint event, so that the wheel marker will show. It’s
157  // not enough to hope setColorFromWidgetCoordinates() would do this,
158  // because setColorFromWidgetCoordinates() would not update the
159  // widget if the mouse click was done at the same position as the
160  // current color marker.
161  update();
162  } else {
163  // Make sure default behaviour like drag-window in KDE’s
164  // “Breeze” widget style works if this widget does not
165  // actually react itself on a mouse event.
166  QWidget::mousePressEvent(event);
167  }

268 void ChromaHueDiagram::paintEvent(QPaintEvent* event)
269 {
270  Q_UNUSED(event);
271 
272  // We do not paint directly on the widget, but on a QImage buffer first:
273  // Render anti-aliased looks better. But as Qt documentation says:
274  //
275  // “Renderhints are used to specify flags to QPainter that may or
276  // may not be respected by any given engine.”
277  //
278  // Painting here directly on the widget might lead to different
279  // anti-aliasing results depending on the underlying window system. This
280  // is especially problematic as anti-aliasing might shift or not a pixel
281  // to the left or to the right. So we paint on a QImage first. As QImage
282  // (at difference to QPixmap and a QWidget) is independant of native
283  // platform rendering, it guarantees identical anti-aliasing results on
284  // all platforms. Here the quote from QPainter class documentation:
285  //
286  // “To get the optimal rendering result using QPainter, you should
287  // use the platform independent QImage as paint device; i.e. using
288  // QImage will ensure that the result has an identical pixel
289  // representation on any platform.”
290  QImage paintBuffer(m_diameter, m_diameter, QImage::Format_ARGB32);
291  paintBuffer.fill(Qt::transparent);
292  QPainter painter(&paintBuffer);
293 
294  QPen pen;
295 
296  // Paint the background.
297  cmsCIELCh backgroundColor;
298  backgroundColor.L = 50;
299  backgroundColor.C = 0;
300  backgroundColor.h = 0;
301  QBrush brush(m_rgbColorSpace->colorRgbBound(backgroundColor));
302  painter.setBrush(brush);
303  pen.setStyle(Qt::PenStyle::NoPen);
304  painter.setPen(pen);
305  painter.setRenderHint(QPainter::Antialiasing, true);
306  painter.drawEllipse(
307  border, // x
308  border, // y
309  m_diameter - 2 * border, // width
310  m_diameter - 2 * border // height
311  );
312  painter.setRenderHint(QPainter::Antialiasing, false);
313  // Paint the diagram itself as available in the cache.
314  updateDiagramCache();
315  painter.drawImage(0, 0, m_diagramImage);
316 
317  // Paint a thin color wheel for better orientation
318  updateWheelCache();
319  painter.drawImage(
320  0,
321  0,
322  m_wheelImage
323  );
324 
325  // paint also an additional marker indicating the hue
326  if (m_isMouseEventActive) {
327  qreal radius =
328  m_diameter / static_cast<qreal>(2) - 2 * markerThickness;
329  // get widget coordinates for our marker
330  QPointF myMarkerInner = PolarPointF(
331  radius - 4 * markerThickness,
332  m_color.toLch().h
333  ).toCartesian();
334  QPointF myMarkerOuter = PolarPointF(
335  radius,
336  m_color.toLch().h
337  ).toCartesian();
338  myMarkerInner.ry() *= -1;
339  myMarkerOuter.ry() *= -1;
340  myMarkerInner += QPointF(m_diagramOffset, m_diagramOffset);
341  myMarkerOuter += QPointF(m_diagramOffset, m_diagramOffset);
342  // draw the line
343  pen = QPen();
344  pen.setWidth(markerThickness);
345  pen.setCapStyle(Qt::FlatCap);
346  pen.setColor(Qt::black);
347  painter.setPen(pen);
348  painter.setRenderHint(QPainter::Antialiasing, true);
349  painter.drawLine(myMarkerInner, myMarkerOuter);
350  }
351 
352  /* Paint a focus indicator.
353  *
354  * We could paint a focus indicator (round or rectangular) around the
355  * marker. Depending on the currently selected hue for the diagram, it
356  * looks ugly because the colors of focus indicator and diagram do not
357  * harmonize, or it is mostly invisible the the colors are similar. So
358  * this apporach does not work well.
359  *
360  * It seems better to paint a focus indicator for the whole widget.
361  * We could use the style primitives to paint a rectangular focus
362  * indicator around the whole widget:
363  * style()->drawPrimitive(QStyle::PE_FrameFocusRect, &option, &painter, this);
364  * However, this does not work well because the chroma-lightness diagram
365  * has usually a triangular shape. The style primitive, however, often
366  * paints just a line at the bottom of the widget. That does not look
367  * good. An alternative approach is that we paint ourself a focus
368  * indicator only on the left of the diagram (which is the place of
369  * black/grey/white, so the won't be any problems with non-harmonic
370  * colors). We can get the color from the current style:
371  * QColor hightlightColor = palette().color(QPalette::Highlight);
372  * QBrush highlightBrush = palette().highlight();
373  * Then we have to design the line that we want to display. It is better
374  * to do that ourself instead of relying on generic QStyle::PE_Frame or
375  * similar solutions as their result seems to be quite unpredictible
376  * accross various styles. So we use markerThickness as line width and
377  * paint it at the left-most possible position. As the border() property
378  * accomodates also to markerRadius, the distance of the focus line to
379  * the real diagram also does, which looks nice.
380  */
381  if (hasFocus()) {
382  painter.setRenderHint(QPainter::Antialiasing);
383  pen = QPen();
384  pen.setWidth(markerThickness);
385  pen.setColor(
386  palette().color(QPalette::Highlight)
387  );
388  painter.setPen(pen);
389  brush = QBrush(Qt::transparent);
390  painter.setBrush(brush);
391  painter.drawEllipse(
392  QPoint(m_diagramOffset + 1, m_diagramOffset + 1), // center
393  (m_diameter - markerThickness) / 2, // x radius
394  (m_diameter - markerThickness) / 2 // y radius
395  );
396  }
397 
398  // Paint the marker on-the-fly.
399  painter.setRenderHint(QPainter::Antialiasing);
400  QPoint imageCoordinates = imageCoordinatesFromColor();
401  pen = QPen();
402  pen.setWidth(markerThickness);
403  // Set color of the marker: Black or white, depending on the lightness of
404  // the currently selected color.
405  if (m_color.toLch().L >= 50) { // range: 0..100
406  pen.setColor(Qt::black);
407  } else {
408  pen.setColor(Qt::white);
409  }
410  painter.setPen(pen);
411  brush = QBrush(Qt::transparent);
412  painter.setBrush(brush);
413  painter.drawEllipse(
414  imageCoordinates.x() - markerRadius,
415  imageCoordinates.y() - markerRadius,
416  2 * markerRadius + 1,
417  2 * markerRadius + 1
418  );
419  PolarPointF lineEndPolar = PolarPointF(
420  QPointF(
421  imageCoordinates.x() - m_diagramOffset,
422  (imageCoordinates.y() - m_diagramOffset) * (-1)
423  )
424  );
425  qreal newRadial = lineEndPolar.radial() - markerRadius - 1.5;
426  QPointF lineEndCartesian;
427  if (newRadial > 0) {
428  lineEndCartesian = PolarPointF(
429  newRadial,
430  lineEndPolar.angleDegree()
431  ).toCartesian();
432  lineEndCartesian.setY(lineEndCartesian.y() * (-1));
433  lineEndCartesian += QPointF(m_diagramOffset, m_diagramOffset);
434  painter.drawLine(
435  QPointF(m_diagramOffset, m_diagramOffset),
436  lineEndCartesian
437  );
438  }
439 
440  // Paint the buffer to the actual widget
441  QPainter(this).drawImage(0, 0, paintBuffer);
442 }

444 void ChromaHueDiagram::wheelEvent(QWheelEvent* event)
445 {
446  if (
447  /* Do nothing while mouse mouvement is tracked anyway. This would be confusing. */
448  (!m_isMouseEventActive) &&
449  /* Only react on wheel events when its in the appropriate place */
450  // TODO Wouldn't it be better to accept wheel events in the whole gray circle, even if outside the gamut?
451  areImageCoordinatesWithinDiagramSurface(event->pos()) &&
452  /* Only react on good old vertical wheels, and not on horizontal wheels */
453  (event->angleDelta().y() != 0)
454  ) {

456  lch.h += Helper::standardWheelSteps(event) * singleStepHue;
457  setColor(
458  FullColorDescription(
459  m_rgbColorSpace,
460  lch,
461  FullColorDescription::outOfGamutBehaviour::sacrifyChroma
462  )
463  );
464  } else {
465  event->ignore();
466  }

485 void ChromaHueDiagram::keyPressEvent(QKeyEvent *event)
486 {
487  // TODO The choise of keys does not seem to be very intuitive
488  cmsCIELCh lch = m_color.toLch();
489  switch (event->key()) {
490  case Qt::Key_Up:
491  lch.C += singleStepChroma;
492  break;
493  case Qt::Key_Down:
494  lch.C -= singleStepChroma;
495  break;
496  case Qt::Key_Left:
497  lch.h += singleStepHue;
498  break;
499  case Qt::Key_Right:
500  lch.h -= singleStepHue;
501  break;
502  case Qt::Key_PageUp:
503  lch.C += pageStepChroma;
504  break;
505  case Qt::Key_PageDown:
506  lch.C -= pageStepChroma;
507  break;
508  case Qt::Key_Home:
509  lch.h += pageStepHue;
510  break;
511  case Qt::Key_End:
512  lch.h -= pageStepHue;
513  break;
514  default:
515  /* Quote from Qt documentation:
516  *
517  * If you reimplement this handler, it is very important that you call the base class
518  * implementation if you do not act upon the key.
519  *
520  * The default implementation closes popup widgets if the user presses the key sequence
521  * for QKeySequence::Cancel (typically the Escape key). Otherwise the event is ignored,
522  * so that the widget's parent can interpret it.
523  */
524  QWidget::keyPressEvent(event);
525  return;
526  }
527  // Here we reach only if the key has been recognized. If not, in the default branch of the
528  // switch statement, we would have passed the keyPressEvent yet to the parent and returned.
529  if (lch.C < 0) {
530  lch.C = 0;
531  }
532  setColor(
533  FullColorDescription(
534  m_rgbColorSpace,
535  lch,
536  FullColorDescription::outOfGamutBehaviour::sacrifyChroma
537  )
538  );
539 }

561 {
562  const qreal scaleFactor =
563  (m_diameter - 2 * border) / static_cast<qreal>(2 * m_maxChroma);
564  return QPoint(
565  qRound((m_color.toLab().a * scaleFactor + m_diagramOffset)),

580 bool ChromaHueDiagram::areImageCoordinatesWithinDiagramSurface(
581  const QPoint imageCoordinates
582 )
583 {
584  qreal radialPixel = PerceptualColor::PolarPointF(
585  imageCoordinates - QPoint(m_diagramOffset, m_diagramOffset)

587  return (
588  radialPixel <= ( (m_diameter - 1) / static_cast<qreal>(2) )

593 void ChromaHueDiagram::setColor(const FullColorDescription &newColor)
594 {
595  if (newColor == m_color) {
596  return;
597  }
598 
599  FullColorDescription oldColor = m_color;
600  m_color = newColor;
601 
602  // update if necessary, the diagram
603  if (m_color.toLch().L != oldColor.toLch().L) {
604  m_diagramCacheReady = false;
605  }
606 
607  // Schedule a paint event:
608  update();
609  Q_EMIT colorChanged(newColor);

618 void ChromaHueDiagram::resizeEvent(QResizeEvent* event)
619 {
620  Q_UNUSED(event);
621  // Find the smaller one of width and height
622  int newDiameter = qMin(size().width(), size().height());
623  // Make sure the diameter is an odd integer (so that center of the
624  // coordinate system is exactly at a pixel, and not between two
625  // pixels.
626  if ((newDiameter % 2) == 0) {
627  newDiameter -= 1;
628  }
629  if (newDiameter < 0) {
630  newDiameter = 0;
631  }
632  // Update the widget
633  if (newDiameter != m_diameter) {
634  m_diameter = newDiameter;
635  m_diagramOffset = (m_diameter - 1) / 2;
636  m_diagramCacheReady = false;
637  m_isWheelCacheReady = false;
638  // As by Qt documentation: The widget will be erased and receive a
639  // paint event immediately after processing the resize event. No
640  // drawing need be (or should be) done inside this handler.
641  }
642 }

685 // How to make sure the diagram has at least a few pixels?
686 QImage ChromaHueDiagram::generateDiagramImage2(
687  const RgbColorSpace *colorSpace,
688  const int imageSize,
689  const qreal maxChroma,
690  const qreal lightness,
691  const int border
692 )
693 {
694  Q_UNUSED(colorSpace);
695 /*
696 Re: [Lcms-user] Out-of-gamut colours?
697 From: Marti <marti@li...> - 2005-07-29 08:23:37
698 
699 Hi,
700 
701 There is a special built-in profile called the null profile. This
702 profile operates on gray scale and returns always zero. Since the gamut
703 warning bypasses the output profile, you can use the null profile to do
704 a quick gamut check: You will obtain 0 for in-gamut and 255 (or whatever
705 gamut alarm you set) for out of gamut.
706 
707 Example:
708 
709 You want to check gamut colors on cmyk.icc, then:
710 
711  cmsSetAlarmCodes(255, 255, 255) // This would turn gray alarm code to 255
712  hNULL = cmsCreateNULLProfile();
713  cmsCreateProofingTransform(
714  hXYZ,
715  TYPE_XYZ_DBL,
716  hNULL, TYPE_GRAY_8,
717  hCMYK,
718  INTENT_RELATIVE_COLORIMETRIC,
719  INTENT_RELATIVE_COLORIMETRIC,
720  (cmsFLAGS_GAMUTCHECK|cmsFLAGS_SOFTPROOFING)
721  );
722 
723 Then, for checking a single color
724 
725  cmsCIEXYZ XYZ;
726  BYTE gamut;
727 
728 ... put value to check into XYZ ...
729 
730  cmsDoTransform(xform, &XYZ, &gamut, 1);
731 */
732  QElapsedTimer myTimer;
733  myTimer.start();
734 
735  /* We do not call cmsSetAlarmCodes(alarmCodes); because this seems to be
736  * not able to change the alpa channel. And apparently, with gamut
737  * checking, transparency seems to be the default for out-of-gamut
738  * colors anyway, so no need to do anything. Anyway, here is some
739  * example code snipped that could change the alarm code. */
740  // cmsUInt16Number alarmCodes[cmsMAXCHANNELS];
741  // cmsUInt16Number alarmCode = std::numeric_limits<cmsUInt16Number>::max();
742  // for (int i = 0; i < cmsMAXCHANNELS; ++i) {
743  // alarmCodes[i] = alarmCode;
744  // }
745  // cmsSetAlarmCodes(alarmCodes);
746 
747  // There is a special built-in profile called the null profile. This
748  // profile operates on gray scale and returns always zero. Since the gamut
749  // warning bypasses the output profile, you can use the null profile to do
750  // a quick gamut check: You will obtain 0 for in-gamut and 255 (or
751  // whatever gamut alarm you set) for out of gamut. Currently we do not use
752  // this technique.
753  // cmsHPROFILE hNULL = cmsCreateNULLProfile();
754  cmsHPROFILE labProfileHandle = cmsCreateLab4Profile(NULL);
755  cmsHPROFILE rgbProfileHandle = cmsCreate_sRGBProfile();
756 
757  cmsHTRANSFORM xform = cmsCreateProofingTransform(
758  labProfileHandle, // Input profile handle
759  TYPE_Lab_DBL, // InputFormat
760  rgbProfileHandle, // Output
761  TYPE_BGRA_8, // OutputFormat TODO endianess?
762  rgbProfileHandle, // Proofing
763  INTENT_ABSOLUTE_COLORIMETRIC, // Intent
764  INTENT_ABSOLUTE_COLORIMETRIC, // ProofingIntent
765  (cmsFLAGS_SOFTPROOFING|cmsFLAGS_GAMUTCHECK) // dwFlags // TODO cmsFLAGS_SOFTPROOFING slown down. And cmsFLAGS_GAMUTCHECK slows down a lot and is not exact.
766  );
767 
768  qDebug()
769  << "Generating the lcms transform took"
770  << myTimer.restart()
771  << "ms.";
772 
773  const int maxIndex = imageSize - 1;
774  // Test if image size is too small.
775  if (maxIndex < 1) {
776  // maxIndex must be at least >= 1 for our algorithm. If they are 0, this would crash (division by 0). // TODO how to solve this?
777  return QImage();
778  }
779 
780  // Setup
781  cmsCIELab lab; // uses cmsFloat64Number internally
782  int x;
783  int y;
784  const qreal scaleFactor =
785  static_cast<qreal>(2 * maxChroma) / (imageSize - 2 * border);
786 
787  int myBufferSize = imageSize * imageSize * 3;
788  double *myBuffer = new double[myBufferSize];
789 
790  // Paint the gamut.
791  lab.L = lightness;
